672 REAL, PARAMETER :: ozcol_85km = 6.7e13

716  THEN

718  ozonecol = ozonecol + ozcol_85km

726  cmessage)
727 END IF
728 
729 IF (lhook) CALL dr_hook('UKCA_STRAT_UPDATE:UKCA_CALC_OZONECOL',zhook_out, &
730  zhook_handle)
731 RETURN
