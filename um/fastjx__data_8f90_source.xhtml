456  REAL FUNCTION flint (tint, t1, t2, t3, f1, f2, f3)
457  IMPLICIT NONE
458 
459  REAL, INTENT(IN) :: tint
460  REAL, INTENT(IN) :: t1
461  REAL, INTENT(IN) :: t2
462  REAL, INTENT(IN) :: t3
463  REAL, INTENT(IN) :: f1
464  REAL, INTENT(IN) :: f2
465  REAL, INTENT(IN) :: f3
466 
467  INTEGER(KIND=jpim), PARAMETER :: zhook_in = 0
468  INTEGER(KIND=jpim), PARAMETER :: zhook_out = 1
469  REAL(KIND=jprb) :: zhook_handle
470 
471  IF (lhook) CALL dr_hook('FASTJX_DATA.FLINT',zhook_in,zhook_handle)
472 
473  IF (tint <= t2) THEN
474  IF (tint <= t1) THEN
475  flint = f1
476  ELSE
477  flint = f1 + (f2 - f1)*(tint - t1)/(t2 -t1)
478  END IF
479  ELSE
480  IF (tint >= t3) THEN
481  flint = f3
482  ELSE
483  flint = f2 + (f3 - f2)*(tint - t2)/(t3 -t2)
484  END IF
485  ENDIF
486 
487  IF (lhook) CALL dr_hook('FASTJX_DATA.FLINT',zhook_out,zhook_handle)
488  RETURN
489  END FUNCTION flint
