262 ! Description:
263 ! Subroutine to apply logic controls and set control variables based on the! options selected in the run_stochastic namelist.
264 
265 ! Dr Hook Modules
266 USE yomhook, ONLY: lhook, dr_hook
267 USE parkind1, ONLY: jprb, jpim
268 USE ereport_mod, ONLY : ereport
269 
270 USE umprintmgr, ONLY : umprint, ummessage
271 IMPLICIT NONE
272 
273 INTEGER :: icode ! error code for ereport
274 ! local temporary arrays
275 CHARACTER(LEN=256) :: cmessage ! out error message
276 CHARACTER(LEN=*), PARAMETER :: routinename='check_run_stochastic'
277 
278 INTEGER(KIND=jpim), PARAMETER :: zhook_in = 0
279 INTEGER(KIND=jpim), PARAMETER :: zhook_out = 1
280 REAL(KIND=jprb) :: zhook_handle
281 
282 IF (lhook) CALL dr_hook('CHECK_RUN_STOCHASTIC',zhook_in,zhook_handle)
283 
284 ! Set random seed logicals based on integer choice in namelist
285 IF (stphseed == 0) THEN
286  l_stphseed_read = .false.
287  l_stphseed_write = .false. ELSE IF (stphseed == 1) THEN
288  l_stphseed_read = .true.
289  l_stphseed_write = .false. ELSE IF (stphseed == 2) THEN
290  l_stphseed_read = .false.
291  l_stphseed_write = .true. END IF
292  ! Add a check to see if biharmonic and smagorinsky dissipation! are both activated, if they are, stop the model.
293 IF (l_skeb2_biharm .AND. l_skeb2_psisdisp ) THEN
294  WRITE(ummessage,'(A)')' ********************************** '
295  CALL umprint(ummessage,src='stochastic_physics_run_mod')
296  WRITE(ummessage,'(A)')' Smagorinsky and Biharmonic are both ON'
297  CALL umprint(ummessage,src='stochastic_physics_run_mod')
298  WRITE(ummessage,'(A)')'(Please choose only one option in the SKEB2 panel)'
299  CALL umprint(ummessage,src='stochastic_physics_run_mod')
300  icode = 1
301  WRITE (cmessage,'(A)')'SKEB2 Biharmonic and Smagorinsky Dnum are both on'
302  CALL ereport(routinename, icode, cmessage)
303 END IF
304 IF (lhook) CALL dr_hook('CHECK_RUN_STOCHASTIC',zhook_out,zhook_handle)
305 RETURN
306 END SUBROUTINE check_run_stochastic
