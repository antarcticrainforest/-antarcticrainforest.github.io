233 ! Description:
234 ! Find the rate of change of temperature with respect to time at specific! altitudes, taking only longwave radiation into account.
235 !
236 ! Method: Uses find_flux_deriv to find the upward and downward fluxes. The
237 ! rate of change of temperature is proportional to the difference between the
238 ! two. Assumes that moisture content makes negligible difference to heat
239 ! capacity of air.
240 !
241 ! Parameters:
242 ! pressure: Pressure, sampled at various heights
243 ! temperature: Sampled at the same heights as pressure
244 ! density: Ditto
245 ! height: Altitudes of pressure, temperature, density samples
246 ! surface_press: Surface pressure
247 ! surface_temp: Surface temperature
248 ! latitude: Latitude of the column
249 ! temperature_rate: The rate of change of temperature
250 ! surface_lw_flux: The net heat flux from the surface due to LW radiation
251 
252  REAL, INTENT(IN) :: pressure(:), temperature(size(pressure)), &
253  density(size(pressure)), height(size(pressure))

255  REAL, INTENT(OUT) :: temperature_rate(size(pressure))
256  REAL, INTENT(OUT) :: surface_lw_flux
257 
258  ! Derivatives of up/downward longwave flux wrt height
259  REAL :: flux_deriv_down(size(pressure)), flux_deriv_up(size(pressure))
260  ! Upwarda and downward fluxes
261  REAL :: flux_up(size(pressure)), flux_down(size(pressure))

264  IF (lhook) CALL dr_hook(routinenme,zhook_in,zhook_handle)

271  surface_temp, latitude, .false., flux_deriv_down, &

290 ! Description:
291 ! Evaluate the derivative of flux wrt height at a height h0 and with an
292 ! existing flux f.
293 !
294 ! Method:
295 ! Evaluates the derivate of flux wrt optical density, opt.dens. wrt pressure
296 ! and pressure wrt height, and multiplies them.
297 !
298 ! Parameters:
299 ! h0: The height at which to evaluate the derivative
300 ! f : The flux at that height
301 ! params : An array of reals which serves to pass other parameters through
302 ! the Runge-Kutta routines. ! params(1) : Magic number, -2.871000183540128e-19, to make sure
303 ! this array is in the right format
304 ! params(2:n+1) : Temperature, sampled at various heights
305 ! params(n+2:2n+1) : Pressure, sampled at various heights
306 ! params(2n+2:3n+1): Heights at which temperature and pressure are sampled
307 ! params(3n+2) : Direction: +ve for upwards, -ve for downwards
308 ! params(3n+3) : Surface pressure
309 ! params(3n+4) : Total atmospheric optical thickness at this point
310

338  CALL error("Parameters with bad magic number passed to eval_flux_deriv")
339  END IF
340

342  temperature = params( 2 : n+1)
343  pressure = params( n+2 : 2*n+1)
344  height = params(2*n+2 : 3*n+1)
345  direction = params(3*n+2) >= 0
346  surface_pressure = params(3*n+3)
347  thickness = params(3*n+4)
348 
349 
350  ! Find the df/dtau part (tau is optical density)
351  eval_flux_deriv(1) = stefanboltzmann * interp(temperature, h0, height)**4 &
352  - f(1)
353  ! Need slightly different equations for upward and downward fluxes
354  IF (direction) THEN !upward
355  eval_flux_deriv(1) = - eval_flux_deriv(1)
356  END IF

358  p = interp(pressure, h0, height)
359  dtaudp = dtau_dp(thickness, surface_pressure, p)
360  ! Find the derivative of flux with respect to height
361  eval_flux_deriv(1) = eval_flux_deriv(1) * dtaudp * deriv(pressure, h0, height)
362 
363  IF (lhook) CALL dr_hook(routinenme,zhook_out,zhook_handle)
364

424 ! Description:
425 ! Finds the optical thickness between the top of the atmosphere and
426 ! the altitudes at which pressure is evaluated. Note that this is
427 ! not used by this module; eval_flux_deriv uses the derivative of
428 ! optical density directly.
429 !
430 ! Parameters:
431 ! pressure: The pressure, sampled at various heights
