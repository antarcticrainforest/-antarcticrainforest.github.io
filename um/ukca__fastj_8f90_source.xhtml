321  INTEGER, PARAMETER :: sw_band_aer = 4

327  REAL,DIMENSION(1:row_length,1:rows,1:model_levels+1) :: pj

334  ! if we do not initialise qsat to zero before calling qsat_wat
335  ! we get failures when we intialise variables to NaN
336  qsat(:,:,:)=0.
337  DO k=1,model_levels
338  CALL qsat_wat(qsat(1:row_length,1:rows,k), &
339  t(1:row_length,1:rows,k), &
340  p_theta_levels(1:row_length,1:rows,k),theta_field_size)
341  ENDDO
342  rh=q(1:row_length,1:rows,:)*(1.0-qsat)/ &
343  (qsat*(1.0-q(1:row_length,1:rows,:)))
344  rh=min(rh,0.99) ! limit humidity to <= 99%
345  rh=max(rh,0.01) ! limit humidity to >= 1%
346 
347 
348 ! Expand land_fraction
349  land_frac=0.
350  DO l=1,land_points
351  j=(land_index(l)-1)/row_length+1
352  i=land_index(l)-(j-1)*row_length
353  land_frac(i,j)=land_frac_ctile(l)
354  END DO
355 
356  sin_true_latitude = f3_at_u(1:ni,1:nj) / two_omega
357 
358 ! rz in cm
359  rz_3d(:,:,1) = r_theta_levels(1:ni,1:nj,0)*100.
360  rz_3d(:,:,2:ml) = r_rho_levels(1:ni,1:nj,2:ml)*100.
361  rz_3d(:,:,ml+1) = r_theta_levels(1:ni,1:nj,ml)*100.
362

372  sulph_aitk=so4_aitken*4.125*d_mass

431 ! set effective diameter for ice crystals
432 ! Formulae from John Edwards to convert from column densities to

434 
435  d_eff=100.0 ! in microns

457  f=(rh(i,j,k)- &

469  END DO
470  END DO
471  END DO
472  ELSE ! dry aerosol
473  DO k=1,model_levels
474  DO j = 1, rows
475  DO i = 1, row_length
476  od(i,j,k) = &
477  sw_spectrum(1)%aerosol%abs(1,ia,sw_band_aer)+ &
478  sw_spectrum(1)%aerosol%scat(1,ia,sw_band_aer)
479  END DO
480  END DO
481  END DO
482  END IF
483  IF (sw_spectrum(1)%aerosol%type_aerosol(ia) &
484  == ip_accum_sulphate) sulph_accu=sulph_accu*od
485  IF (sw_spectrum(1)%aerosol%type_aerosol(ia) &
486  == ip_aitken_sulphate) sulph_aitk=sulph_aitk*od
487  END IF
488  END DO
489 
490  sulphur = sulph_aitk + sulph_accu
491 
492  IF (lhook) CALL dr_hook('FASTJ_COMPUTE_PHOTJ_VALUES',zhook_out,zhook_handle)
493  RETURN
