436  SUBROUTINE twpice_nc_var_to_scm
437 
438  IMPLICIT NONE
439 
440  INTEGER :: i,j,k
441 
442  !=================================
443  ! Add Surface values to the arrays
444  !=================================
445  file_p(0) = file_p_srf(obs_t0) ! Pressure (Pa)
446  file_t(:,0) = file_ts_air ! Temperature (K)
447  file_q(:,0) = (file_rhs_air(:)/100.0) & ! q mix rat. (g/kg)
448  * file_qs_srf(:)*1000.0
449  file_u(:,0) = file_u_srf(:) ! U-wind (m/s)
450  file_v(:,0) = file_v_srf(:) ! V-wind (m/s)
451 
452  file_omega(:,0) = file_omega_srf(:) ! omega (mb/hr)
453  file_divt(:,0) = file_divt(:,1) ! Hor. temperature adv.
454  file_vertdivt(:,0) = file_vertdivt(:,1) ! Ver. dry static energy(s) adv.
455  file_divq(:,0) = file_divq(:,1) ! Hor. mixing ratio (q) adv.
456  file_vertdivq(:,0) = file_vertdivq(:,1) ! Ver. mixing ratio (q) adv.
457 
458 
459  !===========================================
460  ! Scale/Convert to variables required by SCM
461  !===========================================
462 
463  ! Pressure (Pa)
464  file_p(:) = file_p(:) * 100.0
465 
466  ! Mixing ratio: g/kg --> kg/kg
467  file_q(:,:) = 1.0e-3 * file_q(:,:) ! Note: File comments
468  file_divq(:,:) = 1.0e-3 * file_divq(:,:) ! may be wrong, file
469  file_vertdivq(:,:) = 1.0e-3 * file_vertdivq(:,:) ! specifies kg/kg but
470  ! appear to be in g/kg.
471  WHERE (file_q == -0.0) file_q = 0.0 ! Remove -0.0 values
472  WHERE (file_divq == -0.0) file_divq = 0.0 ! Remove -0.0 values
473  WHERE (file_vertdivq == -0.0) file_vertdivq = 0.0 ! Remove -0.0 values
474 
475  ! Mixing ratio > Specific humidity
476  file_q(:,:) = file_q(:,:) / (file_q(:,:) + 1)
477  file_divq(:,:) = file_divq(:,:) / (file_divq(:,:) + 1)
478  file_vertdivq(:,:) = file_vertdivq(:,:) / (file_vertdivq(:,:) + 1)
479 
480  ! Omega(mb/hr) --> w-wind(m/s) using -Omega*R*T/P
481  ! Assuming surface pressure does not change
482  DO j=1, nt_nc
483  file_w(j,:) = - r * file_t(j,:) &
484  * file_omega(j,:)*(1.0/36.0) &
485  / (file_p(:) * g )
486  END DO
487 
488  WHERE (file_w == -0.0) file_w = 0.0 ! Stop creation of -0.0 values
489 
490  ! Temperature --> Theta
491  ! Assumeing surface pressure does not change
492  DO j=1, nt_nc
493  file_th(j,:) = file_t(j,:) &
494  * ((1.0e+5/file_p(:))**(r/cp))
495  END DO
496 
497 
498  ! Calculate z-profile
499  !====================
500 
501  !=================================================================!
502  ! ASSUMING z-heights of forcing data does not vary significantly, !
503  ! Use initial profile for interpolation on UM grid. !
504  !=================================================================!
505 
506  ! Set Surface height to 0m
507  file_z_prf(0) = 0.0
508 
509  DO k=1, nk_in
510  file_layer_thk(k) = -(r/g) &
511  * (( file_t(obs_t0,k) &
512  + file_t(obs_t0,k-1))/2.0) &
513  * (log( file_p(k) &
514  / file_p(k-1)))
515 
516  file_z_prf(k) = file_z_prf(k-1) + file_layer_thk(k)
517  END DO
518 
519  END SUBROUTINE twpice_nc_var_to_scm
