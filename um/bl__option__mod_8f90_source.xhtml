427 ! Description:
428 ! Subroutine to apply logic checks and set control variables based on the
429 ! options selected in the run_bl namelist.
430 
431 ! Dr Hook Modules
432  USE yomhook, ONLY: lhook, dr_hook
433  USE parkind1, ONLY: jprb, jpim
434 
435  USE ereport_mod, ONLY: ereport
436 
437  IMPLICIT NONE
438 
439  INTEGER :: icode ! used for ereport
440  CHARACTER (LEN=80) :: cmessage ! used for ereport
441  CHARACTER (LEN=*), PARAMETER :: RoutineName = 'check_run_bl'
442 
443  INTEGER(KIND=jpim), PARAMETER :: zhook_in = 0
444  INTEGER(KIND=jpim), PARAMETER :: zhook_out = 1
445  REAL(KIND=jprb) :: zhook_handle
446 
447  IF (lhook) CALL dr_hook('CHECK_RUN_BL',zhook_in,zhook_handle)
448 
449 ! check that a 'valid' bl option is provided
450 
451  IF (i_bl_vn == imdi .OR. &
452  i_bl_vn < 0 .OR. &
453  i_bl_vn > 3 ) THEN
454 ! Set switch for turning on boundary layer code, default false
455  WRITE (cmessage,'(A44,I2)') &
456  'Invalid Boundary Layer vn provided, i_bl_vn=', &
457  i_bl_vn
458  icode = 101
459  CALL ereport(routinename, icode, cmessage)
460  END IF
461 
462  IF (i_bl_vn == i_bl_vn_0) THEN
463  WRITE (cmessage,'(A44)') 'Boundary Layer code not used, l_bl_vn = 0'
464  icode = -100
465  CALL ereport(routinename, icode, cmessage)
466  END IF
467 
468  IF (lhook) CALL dr_hook('CHECK_RUN_BL',zhook_out,zhook_handle)
469  RETURN
470  END SUBROUTINE check_run_bl
