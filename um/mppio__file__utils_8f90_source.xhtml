8  ! Implement basic file operations. File ops originated through this
9  ! module will execute immediately if no IO servers are present or will be
10  ! queue on an IO server with fence semantics if IO servers are active. Fence
11  ! semantics mandate that the operation will take place in-order with respect
12  ! to any other IO server queued operations and the operation will be delayed
13  ! until pending IO ops are completed.
14  !
15  ! Commands should normally be issued via high level interfaces;
16  ! file_copy(<src>,<dest>)
17  ! file_delete(<file>)
18  ! create_directory(<name>)
19  !
20  ! which will be translated into a cannonical form and passed to
21  ! the generic function;
22  ! file_action()
23  ! which in turn calls an implementation in c (portio).
24  ! <error code>=file_op(<cannonical string>)
25  !
26  ! The optional force_local argument specifies that (if .TRUE.) the
27  ! operation will be perfomed locally and immediately if the calling rank is
28  ! rank 0 and the op can be assumed to have been completed on rank 0 upon
29  ! return. Other ranks can not make this assumption.
30  !
31  ! An explicit initialisation on all ranks is required before use.

226  SUBROUTINE file_action(command,force_local)
227  IMPLICIT NONE
228  CHARACTER(LEN=*), INTENT(IN) :: command
229  LOGICAL, OPTIONAL :: force_local
230  LOGICAL :: l_force_local ! localised copy of above
231  INTEGER :: pe,i ! loop counters
232  INTEGER :: error ! return code from c
233  INTEGER :: qHandle ! IOS operation handle
234  INTEGER, EXTERNAL :: file_op ! c method to do the work
235  TYPE(ios_metadata_type), &
236  POINTER :: metadata
237  REAL(KIND=jprb) :: zhook_handle
238 
239  IF (lhook) CALL dr_hook('FILE_UTILS:FILE_ACTION',zhook_in,zhook_handle)
240 
241  l_force_local=.false.
242  IF (PRESENT(force_local)) l_force_local=force_local
243 
244  IF (l_ios_active() .AND. &
245  .NOT. l_io_server .AND. &
246  .NOT. l_force_local) THEN
247 
248  IF (io_server_for_unit(file_op_pseudo_unit) == &
249  ios_no_server) THEN
250  CALL ereport('file_utils:file_action',99, &
251  'Module used before initialisation')
252  END IF
253 
254  IF (model_rank == 0) THEN
255 
256  DO pe=1,ios_server_groups
257  qhandle=ios_init_md(-1*io_servers(pe,1), &
258  ios_no_location, &
259  ios_action_fileop)
260  metadata => ios_attach_metadata(qhandle)
261 
262  ! Command
263  DO i=1,ios_string_max
264  metadata % string(i:i)=' '
265  END DO
266  WRITE(metadata % string,'(A)')command
267  metadata % name_length=len_trim(command)
268  metadata % subtype = -1
269  CALL ios_send(qhandle,hasstring=.true.)
270  END DO
271  END IF
272 
273  ELSE
274  IF (model_rank == 0) THEN
275  WRITE(ummessage,'(A,A)')'MPPIO: file op: ',trim(command)
276  CALL umprint(ummessage,src='mppio_file_utils')
277  error=file_op(command,len_trim(command))
278  IF (error /= 0) THEN
279  CALL ereport('file_utils:file_action',error, &
280  'A file action failed in the C level implementation,'// &
281  ' see prior messages')
282  ELSE
283  CALL umprint('MPPIO: file op completed',src='mppio_file_utils')
284  END IF
285  END IF
286  END IF
