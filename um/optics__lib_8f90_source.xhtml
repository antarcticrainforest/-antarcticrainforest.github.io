18  subroutine m_wat(freq, tk, n_r, n_i)
19  implicit none
20 !
21 ! Purpose:
22 ! compute complex index of refraction of liquid water
23 !
24 ! Inputs:
25 ! [freq] frequency (GHz)
26 ! [tk] temperature (K)
27 !
28 ! Outputs:
29 ! [n_r] real part index of refraction
30 ! [n_i] imaginary part index of refraction
31 !
32 ! Reference:
33 ! Based on the work of Ray (1972)
34 !
35 ! Coded:
36 ! 03/22/05 John Haynes (haynes@atmos.colostate.edu)
37 
38 ! ----- INPUTS -----
39  real*8, intent(in) :: freq,tk
40 
41 ! ----- OUTPUTS -----
42  real*8, intent(out) :: n_r, n_i
43 
44 ! ----- INTERNAL -----
45  real*8 ld,es,ei,a,ls,sg,tm1,cos1,sin1
46  real*8 e_r,e_i
47  real*8 pi
48  real*8 tc
49  complex*16 e_comp, sq
50 
51  tc = tk - 273.15
52 
53  ld = 100.*2.99792458e8/(freq*1e9)
54  es = 78.54*(1-(4.579e-3*(tc-25.)+1.19e-5*(tc-25.)**2 &
55  -2.8e-8*(tc-25.)**3))
56  ei = 5.27137+0.021647*tc-0.00131198*tc**2
57  a = -(16.8129/(tc+273.))+0.0609265
58  ls = 0.00033836*exp(2513.98/(tc+273.))
59  sg = 12.5664e8
60 
61  tm1 = (ls/ld)**(1-a)
62  pi = acos(-1.d0)
63  cos1 = cos(0.5*a*pi)
64  sin1 = sin(0.5*a*pi)
65 
66  e_r = ei + (((es-ei)*(1.+tm1*sin1))/(1.+2*tm1*sin1+tm1**2))
67  e_i = (((es-ei)*tm1*cos1)/(1.+2*tm1*sin1+tm1**2)) &
68  +((sg*ld)/1.885e11)
69 
70  e_comp = dcmplx(e_r,e_i)
71  sq = sqrt(e_comp)
72 
73  n_r = real(sq)
74  n_i = aimag(sq)
75 
76  return
77  end subroutine m_wat
