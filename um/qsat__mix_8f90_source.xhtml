114 ! Compute the factor that converts from sat vapour pressure in a
115 ! pure water system to sat vapour pressure in air, fsubw.
116 ! This formula is taken from equation A4.7 of Adrian Gill's book:
117 ! Atmosphere-Ocean Dynamics. Note that his formula works in terms
118 ! of pressure in mb and temperature in Celsius, so conversion of
119 ! units leads to the slightly different equation used here.
120  fsubw = 1.0 + 1.0e-8 * p(i) * ( 4.5 + &
121  6.0e-4*( t(i) - zerodegc ) * ( t(i) - zerodegc ) )
122  fsubw_p1 = 1.0 + 1.0e-8 * p(i+1) * ( 4.5 + &
123  6.0e-4*( t(i+1) - zerodegc ) * ( t(i+1) - zerodegc ) )
124

126 ! it in qmixs.
127 !
128  tt = max(t_low,t(i))
129  tt = min(t_high,tt)
130  atable = (tt - t_low + delta_t) * r_delta_t
131  itable = atable
132  atable = atable - itable
133 
134  tt_p1 = max(t_low,t(i+1))
135  tt_p1 = min(t_high,tt_p1)
136  atable_p1 = (tt_p1 - t_low + delta_t) * r_delta_t
137  itable_p1 = atable_p1
138  atable_p1 = atable_p1 - itable_p1

141  qmixs(i+1) = (1.0 - atable_p1)*es(itable_p1) + &
142  atable_p1*es(itable_p1+1)
143 ! Multiply by fsubw to convert to saturated vapour pressure in air
144 ! (equation A4.6 of Adrian Gill's book).
