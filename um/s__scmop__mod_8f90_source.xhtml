95  FUNCTION stream(strm)
96 
97  IMPLICIT NONE
98 
99  ! Statement function to encode a stream number into an integer
100  ! The default streams for diagnostics to go to will be 1,2,3,4,5 and 6.
101  ! The following should thus be equal to:
102  !
103  ! Stream(1) [2^0=1] + Stream(2) [2^1=2] + Stream(3) [2^2=4]
104  ! Stream(4) [2^3=8] + Stream(5) [2^4=16] + Stream(6) [2^5=32]
105  ! Total = 63
106  !
107  ! where Stream() is the statement function defined above.
108  ! default is 63 (all)
109 
110  INTEGER :: stream, strm
111 
112  stream = 2**(strm-1)
113 
114  END FUNCTION stream
