297  ! thread zero can call MPI, the writer thread may need to proxy MPI
298  ! commands via this thread. We need to hang around and help him
299  ! clear the queue.
300  IF ( lhook ) CALL dr_hook &
301  (routinename//':TAILSPIN',zhook_in,zhook_handle2)
302 
303  IF ( ios_thread_0_calls_mpi ) THEN
304  DO WHILE (ios_getqueueitems() > 0)
305  CALL ios_aux_thread1_assist()
306  CALL um_sleep(ios_backoff_interval)

309
