664  / (file_p_full(j,:) * g)

670  file_qadv(:,:) = file_qadv(:,:)*86400.0
671  file_tadv(:,:) = file_tadv(:,:)*86400.0
672  file_uadv(:,:) = file_uadv(:,:)*86400.0
673  file_vadv(:,:) = file_vadv(:,:)*86400.0

675  ! Surface fluxes in RACMO netcdf file are defined as a downward flux
676  ! Reverse signs for SCM
677  file_flx_h(:) = -1.0 * file_flx_h(:)
678  file_flx_e(:) = -1.0 * file_flx_e(:)
679

681  !====================
682 
683  !=================================================================!
684  ! ASSUMING z-heights of forcing data does not vary significantly, !
685  ! Use initial profile for interpolation on UM grid. !
686  !=================================================================!
687 
688  ! Calculate height on half levels
689  ! Assume element 1 is surface since it is same pressure as p_srf

694  DO k=2, nk_half_nc
695  file_z_half_prf(k) = -(r/g) * file_t(obs_t0,k-1) &
696  * (log( file_p_half(obs_t0,k) &
697  / file_p_half(obs_t0,k-1))) &
698  + file_z_half_prf(k-1)
699  END DO

705  / file_p_srf(obs_t0)))

711  * (log( file_p_full(obs_t0,k) &
712  / file_p_full(obs_t0,k-1))) &
713  + file_z_full_prf(k-1)
714  END DO

717  obs_bot_nc = file_z_full_prf(1)
718
