162 ! ! Section 1. Copy u into field for those model levels with a
163 ! positive diffusion coefficient.
164 ! Copy coeffs and diffusion order for all active levels.
165 ! n_active_levels = 0
166  max_diffusion_order = 0
167  Do k = 1, model_levels
168  If (diffusion_coefficient(k) > 0.0 .and. &
169  & diffusion_order(k) > 0 )Then
170  n_active_levels = n_active_levels + 1
171  indx(n_active_levels) = k
172  active_diffusion_order(n_active_levels) = diffusion_order(k)
173  active_diff_coeff(n_active_levels) = diffusion_coefficient(k)
174  If (diffusion_order(k) > max_diffusion_order ) &
175  & max_diffusion_order = diffusion_order(k)
176  End If
177  End Do

179  level_base = model_levels - n_active_levels
180 ! Last model-level upon which diffusion is inactive
181 ! ! Section 2. Loop over the order to produce order * del squared
182 ! ! Horizontal_level should be the first flat level and as such
183 ! does not need testing for a sloping surface.
184 ! i.e if level_flat < 2 then
185 ! diffusion behaves as original (along eta surfaces)
186 ! To switch off slope test completetely then
187 ! either hard-wire level_flat = 1 or ensure that the argument
188 ! horizontal level <= wet_model_levels - n_active_levels + 1
189 ! level_flat = 1
190  level_flat = n_active_levels - &
191  & model_levels + horizontal_level

195 !$OMP& PRIVATE(k,kk,j,i) &
196 !$OMP& SHARED(off_x, off_y, rows, row_length, field, u, r_at_u, &
197 !$OMP& timestep, timestep_over_r_squared, mask_i, mask_j, &
198 !$OMP& n_active_levels, indx)

200

203  Do j = 1-off_y, rows+off_y
204  Do i = 1-off_x, row_length+off_x
205  field(i,j,k) = u(i,j,kk)

249  End If

262  mask_j(i,j,ka)=1

265  End If

267  End Do

276 !$OMP& SHARED(order, active_diffusion_order, j0, j1, row_length, &

337 ! ! Section 3. Diffusion increment is field * appropriate sign.
338 ! use previously calculated index to calculate which level of field

340 !$OMP PARALLEL DO DEFAULT(NONE) SCHEDULE(STATIC) &
341 !$OMP& PRIVATE(k,j,i,kk,sign) &
342 !$OMP& SHARED(n_active_levels, indx, diffusion_order, row_length, &

344  Do k = 1, n_active_levels
345  kk = indx(k)
346  sign = (-1) ** (diffusion_order(kk)-1)
347  Do j = j0, j1
348  Do i = 1, row_length
349  r_u(i,j,kk) = r_u(i,j,kk) + field(i,j,k) * sign
350  End Do
351  End Do
352  End Do
353 !$OMP END PARALLEL DO
354
