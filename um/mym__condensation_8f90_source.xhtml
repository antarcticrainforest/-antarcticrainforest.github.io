252  ! Here, qsw and qsi are saturated vapor pressure.

262  / (tl(i, j, k - 1) - tetn3w) )
263  qsi(i, j, k) = e0ci * exp(tetn1i * &

265  / (tl(i, j, k - 1) - tetn3i) )
266  END DO

274  DO i = tdims%i_start, tdims%i_end
275  qsw(i, j, k) = repsilon * qsw(i, j, k) &
276  / p_theta_levels(i, j, k - 1)
277  qsi(i, j, k) = repsilon * qsi(i, j, k) &
278  / p_theta_levels(i, j, k - 1)
279  END DO
280  END DO
281  END DO
282  ELSE

287  / (p_theta_levels(i, j, k - 1) &
288  - one_minus_epsilon * qsw(i, j, k))
289  qsi(i, j, k) = repsilon * qsi(i, j, k) &
290  / (p_theta_levels(i, j, k - 1) &
291  - one_minus_epsilon * qsi(i, j, k))
292  END DO
293  END DO
294  END DO
295  END IF

297  ! Calculate sgm
298  DO k = 2, tke_levels
299  DO j = tdims%j_start, tdims%j_end
300  DO i = tdims%i_start, tdims%i_end
301  IF (tl(i, j, k -1) >= ttriple) THEN
302  rice(i, j, k) = 0.0
303  ELSE IF (tl(i, j, k - 1) < temp_ice) THEN
304  rice(i, j, k) = 1.0
305  ELSE
306  rice(i, j, k) = (ttriple - tl(i, j, k - 1)) &
307  / (ttriple - temp_ice)
308  END IF
309 
310  hl = (1.0 - rice(i, j, k)) * lc + rice(i, j, k) * ls
311  qsl = (1.0 - rice(i, j, k)) * qsw(i, j, k) &
312  + rice(i, j, k) * qsi(i, j, k)
313 
314  hl_ovr_cp(i, j, k) = hl / cp
315 
316  dqsl = qsl * repsilon * hl / (r * tl(i, j, k - 1) **2)
317 
318  exner(i, j, k) = (p_theta_levels(i, j, k - 1) / pref) ** kappa
319  qmq(i, j, k) = qw(i, j, k - 1) - qsl
320  alp(i, j, k) = 1.0 /(1.0 + dqsl * hl_ovr_cp(i, j, k))
321  bet(i, j, k) = dqsl * exner(i, j, k)
322 
323  t3sq = max(tsq(i, j, k), 0.0)
324  r3sq = max(qsq(i, j, k), 0.0)
325  c3sq = cov(i, j, k)
326  c3sq = sign(min(abs(c3sq), sqrt(t3sq * r3sq)), c3sq)
327 
328  r3sq = r3sq + bet(i, j, k) ** 2 * t3sq &
329  -2.0 * bet(i, j, k) * c3sq
330  alp_qsl = min(alp(i, j, k) * qsl, qw(i, j, k - 1))
331  sgm(i, j, k) = max( &
332  min(0.5 * alp(i, j, k) * sqrt(max(r3sq, 0.0)), &
333  my_sgm_max_fct * alp_qsl), &
334  my_sgm_min_fct * alp_qsl, 1.e-10)
335  END DO
336  END DO
337  END DO

364  END DO
365  END DO
366  END DO

371  ! Calculate the buoyancy parameters vt and vq
372  DO k = 2, tke_levels
373  DO j = tdims%j_start, tdims%j_end
374  DO i = tdims%i_start, tdims%i_end
375  cld(i, j, k) = 0.5 * (1.0 + erf_val(i, j, k))
376  IF(abs(q1(i, j, k)) > 10.0 ) THEN
377  eq1 = 0.0
378  ELSE
379  eq1 = rrp * exp(- 0.5 * q1(i, j, k) ** 2)
380  END IF
381  ! qll = ql / (2 * sgm)
382  qll = max(cld(i, j, k) * q1(i, j, k) + eq1, 0.0)
383 
384  IF(qw(i, j, k) < 1.e-10) THEN
385  ql(i, j, k) = 0.0
386  ELSE
387  ql(i, j, k) = max( &
388  2. * sgm(i, j, k) * qll, 0.0)
389  END IF
390  ! To avoid negative QV (for safety)
391  ql(i, j, k) = min(ql(i, j, k), qw(i, j, k - 1) * 0.5)
392

395  pt_tmp = t(i, j, k - 1) * r_exner

397  - (1.0 + c_virtual) * ql(i, j, k)
398  rac = alp(i, j, k) * (cld(i, j, k) &
399  - qll * eq1) * (q2p * qt - (1.0 + c_virtual) * pt_tmp)
400 
401  vt(i, j, k) = qt - rac * bet(i, j, k)
402  vq(i, j, k) = c_virtual * pt_tmp + rac
403  END DO
404  END DO
405  END DO

434  END DO

495  DO j = tdims%j_start, tdims%j_end
496  DO i = tdims%i_start, tdims%i_end

498  END DO
499  END DO
500  END DO

502  CALL scmoutput(scm_diag, &
503  'Q1','normalized excessive moisture',' ', &

507 #endif
