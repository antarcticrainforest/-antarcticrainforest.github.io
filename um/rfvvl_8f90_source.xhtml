86  IF (lhook) CALL dr_hook('RFVVL',zhook_in,zhook_handle)
87  z=firstrow ! N.B. THIS IS CO-LATITUDE
88  DO jrow=1,n_rows ! CALCULATE ROTATION COEFFICIENTS
89  e=cos(z)*dlong
90  c_rot(jrow)=cos(e)
91  s_rot(jrow)=sin(e)
92  z=z+dlat
93  ENDDO ! JROW
94 !
95 -- calculate coeffs for e->w sweep(also used for w->e)
96  DO jrow=1,n_rows
97  z=npass*(dlong*coslat(jrow))**2*0.25
98  DO jpt=1,row_length
102  e=z/scale(jpt,jrow)**2
103  a_we(jpt,jrow)= 1.+e-sqrt(e*(e+2.))
104  ENDDO ! JPT
105  ENDDO ! JROW
106 !
107 -- calculate coeffs for n->s sweep(also used for s->n)
108  z=npass*dlat**2*0.25
109  DO jrow=1,n_rows
110  DO jpt=1,row_length
111 ! STRICTLY, SCALE SHOULD BE AVERAGE JROW & JROW+(-)1
112 ! NOT DOING SO MAKES THE CODE SIMPLER.
113 ! SINCE SCALE IS SMOOTH, THE DIFFERENCE IS NEGLIGABLE.
114  e=z/scale(jpt,jrow)**2
115  a_ns(jpt,jrow)= 1.+e-sqrt(e*(e+2.))
116  ENDDO ! JPT
117  ENDDO ! JROW

119 !L
120 !L*** START LOOP OVER PASSES
121  DO jpass=1,npass ! ===========================================
122 !L
123 !L*** FILTER W->E -!L
124 !
125  IF(m_grid == 0)THEN ! BCS SPECIFIED TO ALLOW FOR CONSTANT OUTSIDE
126 ! DO W BC
127 ! THE BOUNDARY CONDITIONS DEPEND ON THE NUMBER OF PREVIOUS PASSES
128 ! IN THE OPPOSITE DIRECTION.
129 ! THE FORMULAE FOR 0,1,2 PREVIOUS PASSES ARE:! G1=(1-A)F1
130 ! G1=(1/(1+A))F1
131 ! G1=(1/(1+A)(1-A**2))(F1-A**3*F2)
132 ! SEE METO11 WORKING PAPER 91 BY R J PURSER (1987) FOR DETAILS
133  IF (jpass == 1)THEN ! NO PREVIOUS E->W PASS
134  DO jrow=1,n_rows
135  u(1,jrow)=u(1,jrow)*(1.-a_we(1,jrow))
136  v(1,jrow)=v(1,jrow)*(1.-a_we(1,jrow))
137  zu(jrow)=u(1,jrow)
138  zv(jrow)=v(1,jrow)
139  ENDDO ! JROW
140  ELSE IF(jpass == 2)THEN ! ONE PREVIOUS E->W PASS
141  DO jrow=1,n_rows
142  u(1,jrow)=u(1,jrow)/(1.+a_we(1,jrow))
143  v(1,jrow)=v(1,jrow)/(1.+a_we(1,jrow))
144  zu(jrow)=u(1,jrow)
145  zv(jrow)=v(1,jrow)
146  ENDDO ! JROW
147  ELSE IF(jpass >= 3)THEN ! TWO PREVIOUS E->W PASSES
148 ! BC FOR 2 IS ALSO USED FOR >2
149  DO jrow=1,n_rows ! EQN 6.5.28
150  u(1,jrow)=(u(1,jrow)-a_we(1,jrow)**3*u(2,jrow)) &
151  & /((1.+a_we(1,jrow))*(1.-a_we(1,jrow)**2))
152  v(1,jrow)=(v(1,jrow)-a_we(1,jrow)**3*v(2,jrow)) &
153  & /((1.+a_we(1,jrow))*(1.-a_we(1,jrow)**2))
154  zu(jrow)=u(1,jrow)
155  zv(jrow)=v(1,jrow)
156  ENDDO ! JROW
157  ENDIF ! JPASS
158 ! ELSE ! M_GRID /= 0 NO SPECIAL TREATMENT OF BOUNDARY
159  ENDIF ! M_GRID
160 !
161 !L FILTER W->E PASS
162  DO jpt=2,row_length

165  & (c_rot(jrow)*zu(jrow)+s_rot(jrow)*zv(jrow)-u(jpt,jrow))
166  v(jpt,jrow)=v(jpt,jrow)+a_we(jpt,jrow)* &
167  & (c_rot(jrow)*zv(jrow)-s_rot(jrow)*zu(jrow)-v(jpt,jrow))
168  zu(jrow)=u(jpt,jrow)
169  zv(jrow)=v(jpt,jrow)
170  ENDDO !)) JROW
171  ENDDO !)) JPT
172 !L
173 !L*** FILTER E->W -!L
174  IF(m_grid == 0)THEN ! BCS SPECIFIED TO ALLOW FOR CONSTANT OUTSIDE
175 ! DO E BC
176  IF (jpass == 1)THEN ! ONE PREVIOUS W->E PASS
177  DO jrow=1,n_rows
178  u(row_length,jrow)=u(row_length,jrow) &
179  & /(1.+a_we(row_length,jrow))
180  v(row_length,jrow)=v(row_length,jrow) &
181  & /(1.+a_we(row_length,jrow))
182  zu(jrow)=u(row_length,jrow)
183  zv(jrow)=v(row_length,jrow)
184  ENDDO ! JROW
185  ELSE IF(jpass >= 2)THEN ! TWO PREVIOUS W->E PASSES

189  & a_we(row_length,jrow)**3*u(row_length-1,jrow)) &
190  & /((1.+a_we(row_length,jrow))*(1.-a_we(row_length,jrow)**2))
191  v(row_length,jrow)=(v(row_length,jrow)- &
192  & a_we(row_length,jrow)**3*v(row_length-1,jrow)) &
193  & /((1.+a_we(row_length,jrow))*(1.-a_we(row_length,jrow)**2))
194  zu(jrow)=u(row_length,jrow)
195  zv(jrow)=v(row_length,jrow)
196  ENDDO ! JROW
197  ENDIF ! JPASS
198 ! ELSE ! M_GRID /= 0 NO SPECIAL TREATMENT OF BOUNDARY
199  ENDIF ! M_GRID

202  DO jpt=row_length-1,1,-1
203  DO jrow=1,n_rows

206  v(jpt,jrow)=v(jpt,jrow)+a_we(jpt,jrow)* &

208  zu(jrow)=u(jpt,jrow)
209  zv(jrow)=v(jpt,jrow)
210  ENDDO !)) JROW
211  ENDDO !)) JPT

215 ! DO N BC
216  IF (jpass == 1)THEN ! NO PREVIOUS S->N PASS
217  DO jpt=1,row_length
218  u(jpt,1)=u(jpt,1)*(1.-a_ns(jpt,1))
219  v(jpt,1)=v(jpt,1)*(1.-a_ns(jpt,1))
220  ENDDO ! JPT
221  ELSE IF(jpass == 2)THEN ! ONE PREVIOUS S->N PASS
222  DO jpt=1,row_length
223  u(jpt,1)=u(jpt,1)/(1.+a_ns(jpt,1))
224  v(jpt,1)=v(jpt,1)/(1.+a_ns(jpt,1))
225  ENDDO ! JPT
226  ELSE IF(jpass >= 3)THEN ! TWO PREVIOUS S->N PASSES
227 ! BC FOR 2 IS ALSO USED FOR >2

231  v(jpt,1)=(v(jpt,1)-a_ns(jpt,1)**3*v(jpt,2)) &
232  & /((1.+a_ns(jpt,1))*(1.-a_ns(jpt,1)**2))
233  ENDDO ! JPT
234  ENDIF ! JPASS
235 ! ELSE ! M_GRID /= 0 NO SPECIAL TREATMENT OF BOUNDARY
236  ENDIF ! M_GRID
237 !

239  DO jrow=2,n_rows

244  & (v(jpt,jrow-1)-v(jpt,jrow))*a_ns(jpt,jrow)
245  ENDDO !)) JPT
246  ENDDO !)) JROW

249  IF(m_grid == 0)THEN ! BCS SPECIFIED TO ALLOW FOR CONSTANT OUTSIDE
250 ! DO S BC
251  IF (jpass == 1)THEN ! ONE PREVIOUS N->S PASS
252 ! BC FOR AFTER A SINGLE SWEEP IS USED FOR ALL LATER SWEEPS
253  DO jpt=1,row_length
254  u(jpt,n_rows)=u(jpt,n_rows)/(1.+a_ns(jpt,n_rows))
255  v(jpt,n_rows)=v(jpt,n_rows)/(1.+a_ns(jpt,n_rows))
256  ENDDO ! JPT
257  ELSE IF(jpass >= 2)THEN ! TWO PREVIOUS S->N PASSES
258 ! BC FOR 2 IS ALSO USED FOR >2
259  DO jpt=1,row_length ! EQN 6.5.28
260  u(jpt,n_rows)= &
261  & (u(jpt,n_rows)-a_ns(jpt,n_rows)**3*u(jpt,n_rows-1)) &
262  & /((1.+a_ns(jpt,n_rows))*(1.-a_ns(jpt,n_rows)**2))
263  v(jpt,n_rows)= &
264  & (v(jpt,n_rows)-a_ns(jpt,n_rows)**3*v(jpt,n_rows-1)) &
265  & /((1.+a_ns(jpt,n_rows))*(1.-a_ns(jpt,n_rows)**2))
266  ENDDO ! JPT
267  ENDIF ! JPASS
268 ! ELSE ! M_GRID /= 0 NO SPECIAL TREATMENT OF BOUNDARY
269  ENDIF ! M_GRID
270 !
271 !L FILTER N->S PASS
272  DO jrow=n_rows-1,1,-1
273  DO jpt=1,row_length
274  u(jpt,jrow)=u(jpt,jrow)+ &
275  & (u(jpt,jrow+1)-u(jpt,jrow))*a_ns(jpt,jrow)
276  v(jpt,jrow)=v(jpt,jrow)+ &
277  & (v(jpt,jrow+1)-v(jpt,jrow))*a_ns(jpt,jrow)
278  ENDDO !)) JPT
279  ENDDO !)) JROW
280 !
281  ENDDO ! JPASS ====================================================
282 !
