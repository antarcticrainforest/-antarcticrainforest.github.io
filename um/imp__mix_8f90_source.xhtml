145 ! ! 4. Calculate those matrix and vector elements on the LHS of eqn
146 ! which are to do with implicit solution of the tracer
147 ! transport problem at the surface and between all levels.
148 ! Begin with "upward sweep" through lower half of matrix).

155 ! 4.2 Lowest atmospheric layer FIELD row of matrix.
156 
157  ! "Explicit" increment to FIELD(1)
158  d_field(i,j,1) = -dtrdz(i,j,1) * &
159  ( f_field(i,j,2) - f_field(i,j,1) &
160  - surf_dep_flux(i,j) )

164  rbf = 1.0 / ( 1.0 - af(i,j,1) - cf )

258  END DO
259  END DO
260  END DO
261 !$OMP END DO
262

264  DO j = pdims%j_start, pdims%j_end
265  DO i = pdims%i_start, pdims%i_end
266  rr_sq = 1.0 / ( r_theta_levels(i,j,0)*r_theta_levels(i,j,0) )
267  gamma_rhok_dep(i,j) = rr_sq * gamma_rhok_dep(i,j)
268  f_field(i,j,1) = rr_sq * f_field(i,j,1)
269  surf_dep_flux(i,j) = rr_sq * surf_dep_flux(i,j)
270  END DO
271  END DO
272 !$OMP END DO NOWAIT

280  DO i = pdims%i_start, pdims%i_end
281  surf_dep_flux(i,j) = surf_dep_flux(i,j) &

283  END DO
284  END DO

289 
290

292  DO k = 2, bl_levels
293  km1 = k-1
294  DO j = pdims%j_start, pdims%j_end
295  DO i = pdims%i_start, pdims%i_end
296 
297 ! Calculate and store implicit fluxes due to local mixing.
298 
299  f_field(i,j,k) = f_field(i,j,k) - gamma_rhokh_rdz(i,j,k) &
300  * ( d_field(i,j,k) - d_field(i,j,km1) )
301  END DO
302  END DO
303  END DO
