207 SUBROUTINE check_run_ukca()
208 
209 ! Description:
210 ! Subroutine to apply logic checks based on the! options selected in the run_ukca namelist.
211 ! Note that some chemistry scheme specific checks
212 ! are done in ukca_setup_chem
213 
214 USE ukca_photo_scheme_mod, ONLY: i_ukca_nophot, i_ukca_phot2d, &
215  i_ukca_fastj, i_ukca_fastjx use ereport_mod, only: ereport
216 USE parkind1, ONLY: jprb, jpim
217 USE yomhook, ONLY: lhook, dr_hook
218 
219 IMPLICIT NONE
220 
221 CHARACTER (LEN=*), PARAMETER :: routinename = 'check_run_ukca'
222 CHARACTER (LEN=70) :: cmessage ! Error message
223 INTEGER :: errcode ! Variable passed to ereport
224 
225 INTEGER(KIND=jpim), PARAMETER :: zhook_in = 0
226 INTEGER(KIND=jpim), PARAMETER :: zhook_out = 1
227 REAL(KIND=jprb) :: zhook_handle
228 
229 IF (lhook) CALL dr_hook(routinename,zhook_in,zhook_handle)
230 
231 ! Check photolysis switches
232 IF ( i_ukca_photol /= i_ukca_nophot .AND. &
233  i_ukca_photol /= i_ukca_phot2d .AND. &
234  i_ukca_photol /= i_ukca_fastj .AND. &
235  i_ukca_photol /= i_ukca_fastjx ) THEN
236  cmessage='Unknown photolysis scheme.'
237  errcode=abs(i_ukca_photol)
238  CALL ereport(routinename,errcode,cmessage)
239 END IF
240 
241 ! GLOMAP-mode related switches
242 IF ( l_ukca_mode .AND. .NOT. l_ukca ) THEN
243  cmessage='Cannot use GLOMAP-mode aerosols without UKCA'
244  errcode=1
245  CALL ereport(routinename,errcode,cmessage)
246 END IF
247 ! Direct aerosol effects
248 IF ( l_ukca_radaer .AND. .NOT. l_ukca_mode ) THEN
249  cmessage='Cannot use RADAER without GLOMAP-mode aerosols'
250  errcode=2
251  CALL ereport(routinename,errcode,cmessage)
252 END IF
253 ! Indirect aerosol effects
254 IF ( (l_ukca_aie1 .OR. l_ukca_aie2) .AND. .NOT. l_ukca_mode ) THEN
255  cmessage='Cannot use AIE without GLOMAP-mode aerosols'
256  errcode=2
257  CALL ereport(routinename,errcode,cmessage)
258 END IF
259 IF (lhook) CALL dr_hook(routinename,zhook_out,zhook_handle)
260 
261 END SUBROUTINE check_run_ukca
