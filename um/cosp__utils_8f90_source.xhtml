50 SUBROUTINE cosp_precip_mxratio(Npoints,Nlevels,Ncolumns,p,T,prec_frac, &
51  prec_type,n_ax,n_bx,alpha_x,c_x,d_x,g_x,a_x,b_x,gamma1, &
52  gamma2,gamma3,gamma4,flux,mxratio,reff)
53 
54  ! Input arguments, (IN)
55  INTEGER,INTENT(IN) :: Npoints,Nlevels,Ncolumns
56  REAL,INTENT(IN),DIMENSION(Npoints,Nlevels) :: p,T,flux
57  REAL,INTENT(IN),DIMENSION(Npoints,Ncolumns,Nlevels) :: prec_frac
58  REAL,INTENT(IN) :: n_ax,n_bx,alpha_x,c_x,d_x,g_x,a_x,b_x,gamma1,gamma2, &
59  gamma3,gamma4,prec_type
60  ! Input arguments, (OUT)
61  REAL,INTENT(OUT),DIMENSION(Npoints,Ncolumns,Nlevels) :: mxratio
62  REAL,INTENT(INOUT),DIMENSION(Npoints,Ncolumns,Nlevels) :: reff
63  ! Local variables
64  INTEGER :: i,j,k
65  REAL :: sigma,one_over_xip1,xi,rho0,rho,lambda_x,gamma_4_3_2,delta
66 
67  mxratio = 0.0
68 
69  ! N_ax is used to control which hydrometeors need to be computed
70  IF (n_ax >= 0.0) THEN
71  xi = d_x/(alpha_x + b_x - n_bx + 1.0)
72  rho0 = 1.29
73  sigma = (gamma2/(gamma1*c_x))*(n_ax*a_x*gamma2)**xi
74  one_over_xip1 = 1.0/(xi + 1.0)
75  gamma_4_3_2 = 0.5*gamma4/gamma3
76  delta = (alpha_x + b_x + d_x - n_bx + 1.0)
77 
78  DO k=1,nlevels
79  DO j=1,ncolumns
80  DO i=1,npoints
81  IF ((prec_frac(i,j,k)==prec_type).OR.(prec_frac(i,j,k)==3.)) THEN
82  rho = p(i,k)/(287.05*t(i,k))
83  mxratio(i,j,k)=(flux(i,k)*((rho/rho0)**g_x)*sigma)**one_over_xip1
84  mxratio(i,j,k)=mxratio(i,j,k)/rho
85  ! Compute effective radius
86  IF ((reff(i,j,k) <= 0.0).AND.(flux(i,k) /= 0.0)) THEN
87  lambda_x = &
88  (a_x*c_x*((rho0/rho)**g_x)*n_ax*gamma1/flux(i,k))**(1./delta)
89  reff(i,j,k) = gamma_4_3_2/lambda_x
90  ENDIF
91  ENDIF
92  ENDDO
93  ENDDO
94  ENDDO
95  ENDIF
96 END SUBROUTINE cosp_precip_mxratio

190  SUBROUTINE z_to_dbz_2d(mdi,z)
191  REAL,INTENT(IN) :: mdi
192  REAL,DIMENSION(:,:),INTENT(INOUT) :: z
193  ! Reflectivity Z:
194  ! Input in [m3]
195  ! Output in dBZ, with Z in [mm6 m-3]
196 
197  ! 1.e18 to convert from [m3] to [mm6 m-3]
198  z = 1.e18*z
199  WHERE (z > 1.0e-6) ! Limit to -60 dBZ

201  ELSEWHERE
202  z = mdi
