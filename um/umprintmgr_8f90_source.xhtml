351 ! ++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
352  RECURSIVE SUBROUTINE umprint(line,level,pe,src,model,stdErrorToo)
353 !$ USE omp_lib
354  IMPLICIT NONE
355 
356  CHARACTER(LEN=*) :: line
357  INTEGER, OPTIONAL :: level
358  INTEGER, OPTIONAL :: pe
359  CHARACTER(LEN=*), OPTIONAL :: src
360  CHARACTER(LEN=*), OPTIONAL :: model
361  LOGICAL, OPTIONAL :: stdErrorToo
362  INTEGER :: i
363  INTEGER :: thread
364  LOGICAL :: willPrint
365  REAL(KIND=jprb) :: zhook_handle
366  CHARACTER(LEN=*), PARAMETER :: RoutineName='umPrint'
367  CHARACTER(LEN=maxLineLen) :: finalBuffer
368 
369  IF (lhook) CALL dr_hook('UMPRINT',zhook_in,zhook_handle)
370 
371  ! Decide whether to print or not
372  willprint=printeractive
373 
374  IF (PRESENT(level)) THEN
375  IF (printstatus < level) willprint=.false.
376  END IF
377 
378  IF (PRESENT(pe)) THEN
379  IF (pe /= mype) willprint=.false.
380  END IF
381 
382  IF (willprint) THEN
383  IF (.NOT.ostreamopen) THEN
384  CALL umprintopenstream()
385  END IF
386 
387  IF ( PRESENT(src) .AND. prnt_src_pref ) THEN
388  ! Prefix the output with src and recall dropping the argument
389  finalbuffer=trim(adjustl(src)) // ':' // trim(line)
390  CALL umprint(trim(finalbuffer), &
391  model=model, &
392  stderrortoo=stderrortoo)
393 
394  ELSE IF ( PRESENT(model) .AND. prnt_src_pref ) THEN
395  ! Prefix the output with model and recall dropping the argument
396  ! src has already been stripped above so no need to pass through.
397  finalbuffer=trim(adjustl(model)) // ':' // trim(line)
398  CALL umprint(trim(finalbuffer), &
399  stderrortoo=stderrortoo)
400 
401  ELSE ! Nothing more to bolt to the string...
402 
403  IF (len_trim(line) > prnt_paper_width) splitcount=splitcount+1
404 
405  IF (len_trim(line) > prnt_paper_width .AND. prnt_split_lines ) THEN
406 
407  CALL umprint("NOTE: Long line has been split", &
408  src='umPrintMgr',level=prdiag)
409  DO i=0,len_trim(line)/prnt_paper_width
410  CALL umprint(line(i*prnt_paper_width+1: &
411  min( (i+1)*prnt_paper_width , len_trim(line) )), &
412  stderrortoo=stderrortoo)
413  END DO
414 
415  ELSE
416  printcount=printcount+1
417 
418  ! If we are in a parallel region, add the thread ID.
419 
420 !$ IF ( omp_in_parallel() ) THEN
421 !$ thread=omp_get_thread_num()
422 !$ finalBuffer='['//TRIM(str_i(thread))//'] '//TRIM(line)
423 !$ ELSE
424  finalbuffer=trim(line)
425 !$ END IF
426  IF (PRESENT(stderrortoo)) THEN
427  IF (stderrortoo) THEN
428  WRITE(estreamunit,'(A)')trim(finalbuffer)
429  END IF
430  END IF
431 
432  WRITE(ostreamunit,'(A)')trim(finalbuffer)
433 
434  ! flush the stream if options suggest to do so
435  IF (prnt_force_flush .OR. printstatus >= prdiag) &
436  CALL umprintflush()
437 
438  END IF
439  END IF
440  END IF ! willPrint
441 
442  IF (lhook) CALL dr_hook('UMPRINT',zhook_out,zhook_handle)
443 
444  END SUBROUTINE umprint
