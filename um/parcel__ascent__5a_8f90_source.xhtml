208  ! Require t_ref on all point for qsat call

222

236 ! Undilute parcel calculation always required
237 
238 !$OMP PARALLEL DEFAULT(SHARED) PRIVATE(ii, lrcp_const, lrcp_const_env, &
239 !$OMP& l_const, dq_sat_env, dq_sat_par, q_liq_parc, q_liq_env, z_pr, &
240 !$OMP& lrcp_const_parc, ql_remove, max_qw_plume, q_vap_parc, th_par, &
241 !$OMP& temp_parc, qcl_parc, qcf_parc, l_const_env, dq_sat_par_dil)
242 
243 !$OMP DO SCHEDULE(STATIC)
244  DO ii=1, npnts
245 
246  If(t_ref(ii) > tm) THEN
247  lrcp_const = lcrcp
248  l_const = lc
249  ELSE
250  lrcp_const = lsrcp
251  l_const = ls
252  END IF
253  If(t(ii,k) > tm) THEN
254  lrcp_const_env = lcrcp
255  l_const_env = lc
256  ELSE
257  lrcp_const_env = lsrcp
258  l_const_env = ls
259  END IF
260 
261  dq_sat_env = repsilon*l_const_env*qsat_env(ii)/(r*t(ii,k)**2)
262  dq_sat_par = repsilon*l_const *qsat_lev(ii)/(r*t_ref(ii)**2)
263 
264  q_liq_parc = max( 0.0, ( qw_plume(ii) - qsat_lev(ii) &
265  -dq_sat_par*( sl_plume(ii)-gamma_dry*z_theta(ii,k)-t_ref(ii) ) &
266  ) / (1.0+lrcp_const*dq_sat_par) )
267 
268  q_liq_env = max( 0.0, ( qw(ii,k) - qsat_env(ii) &
269  -dq_sat_env*( tl(ii,k) - t(ii,k) ) &
270  ) / (1.0+lrcp_const_env*dq_sat_env) )
271 !
272 ! add on the difference in the environment's ql as calculated by the
273 ! UM cloud scheme (using some RH_CRIT value) and what it
274 ! would be If RH_CRIT=1. This THEN imitates partial condensation
275 ! in the parcel.
276 !
277  ql_parc(ii,k) = q_liq_parc + qcl(ii,k) + qcf(ii,k) - q_liq_env
278  t_parc(ii,k)=sl_plume(ii)-gamma_dry*z_theta(ii,k) +lrcp_const*ql_parc(ii,k)
279 
280 ! May need to recalculate if T_parc is > Tm and T_ref < Tm
281 
282  IF (t_ref(ii) <= tm .and. t_parc(ii,k) > tm) THEN
283 
284 ! recalculate using corrected latent heats
285  lrcp_const_parc = lcrcp
286 
287  q_liq_parc = max( 0.0, ( qw_plume(ii) - qsat_lev(ii) &
288  -dq_sat_par*( sl_plume(ii)-gamma_dry*z_theta(ii,k)-t_ref(ii) ) &
289  ) / (1.0+lrcp_const_parc*dq_sat_par) )
290 
291  ql_parc(ii,k) = q_liq_parc + qcl(ii,k) + qcf(ii,k)- q_liq_env
292 ! revised at parcel calculation
293 
294  t_parc(ii,k)=sl_plume(ii)-gamma_dry*z_theta(ii,k) &
295  +lrcp_const_parc*ql_parc(ii,k)
296 
297  END IF ! test on T_ref and t_parc
298 
299  ! Add water removal from undilute plume IF (plume_water_load == 1) THEN
300 
301  ! water removed from parcel after condensation if > 0.001 kg/kg
302  IF (ql_parc(ii,k) > 0.001) THEN
303  ql_remove = ql_parc(ii,k) - 0.001
304  ql_parc(ii,k) = 0.001
305  qw_plume(ii) = qw_plume(ii) - ql_remove
306  ! Also adjust sl_plume as well as altered energy
307  sl_plume(ii) = sl_plume(ii) + lrcp_const*ql_remove
308  END IF
309  ELSE IF (plume_water_load == 2) THEN
310 
311  ! water remaining depends on qsat environment max_qw_plume = fac_qsat*qsat_env(ii)
312  max_qw_plume = min(qlcrit,max_qw_plume) ! max
313  max_qw_plume = max(qlmin,max_qw_plume) ! min value IF (ql_parc(ii,k) > max_qw_plume) THEN
314  ql_remove = ql_parc(ii,k) - max_qw_plume ql_parc(ii,k) = max_qw_plume
315  qw_plume(ii) = qw_plume(ii) - ql_remove
316  ! Also adjust sl_plume as well as altered energy
317  sl_plume(ii) = sl_plume(ii) + lrcp_const*ql_remove
318 
319  END IF
320  END IF ! end test on plume_water_load
321 
322  q_vap_parc=qw_plume(ii)-ql_parc(ii,k)
323  qv_parc(ii,k) = q_vap_parc
324  t_dens_parc(ii,k)=t_parc(ii,k)*(1.0+c_virtual*q_vap_parc-ql_parc(ii,k))
325 
326 
327 ! calculate t_ref for next level
328  IF (k > 1 .and. k < wet_levels-1) THEN
329  z_pr = (z_theta(ii,k+1)-z_theta(ii,k))/(z_theta(ii,k)-z_theta(ii,k-1))
330  th_par = t_parc(ii,k)/exner_theta_levels(ii,k)
331  th_ref(ii) = th_par*(1.+z_pr) - th_par_km1(ii)*z_pr
332 
333 ! Check sensible value otherwise set to previous reference value
334 ! Problems can occur near top of model where calculation are nolonger! important.
335  IF (th_ref(ii) < 0.0) THEN
336  th_ref(ii) = th_par_km1(ii)
337  END IF
338  IF (th_par > 0.0) THEN th_par_km1(ii) = th_par
339  END IF
340  END IF
341 
342 
343 ! Dilute parcel ascent
344 
345  IF (l_dilute) THEN
346  IF (k <= nlcl(ii)) THEN ! Dilute parcel same as undilute
347  ! parcel ascent (no entrainment)
348 
349  sl_parc(ii,k) = sl_plume(ii)
350  qw_parc(ii,k) = qw_plume(ii)
351  t_parc_dil(ii,k) = t_parc(ii,k)
352  ql_parc_dil(ii,k) = ql_parc(ii,k) t_dens_parc_dil(ii,k)= t_dens_parc(ii,k)
353  th_ref_dil(ii) = th_ref(ii)
354  th_par_km_dil(ii) = th_par_km1(ii)
355 
356  ELSE ! Dilute parcel ascent now required
357 
358  If(t_ref_dil(ii) > tm) THEN
359  lrcp_const = lcrcp
360  l_const = lc
361  ELSE
362  lrcp_const = lsrcp
363  l_const = ls
364  END IF
365 
366 
367 ! Dilute parcel
368 
369 ! Mix in entrain_fraction from environmental air from level below and! raise this to current level.
370 ! Assume mix in fraction of mass from environment.
371 ! Estimate parcel properties after mixing air from environment with! parcel. Temperature given approximately by average
372 
373  temp_parc = (t_parc_dil(ii,k-1) &
374  + entrain_fraction(ii,k)*t(ii,k-1)) &
375  /(1.+entrain_fraction(ii,k))
376 
377  qw_parc(ii,k) = (qw_parc(ii,k-1) + &
378  entrain_fraction(ii,k)*qw(ii,k-1)) & /(1.+entrain_fraction(ii,k))
379  qcl_parc = (ql_parc_dil(ii,k-1) + & entrain_fraction(ii,k)*qcl(ii,k-1)) & /(1.+entrain_fraction(ii,k))
380  qcf_parc = (0.0 + & entrain_fraction(ii,k)*qcf(ii,k-1)) & /(1.+entrain_fraction(ii,k))
381 ! All condensed water either ice or liquid based on t_ref sl_parc(ii,k) = temp_parc - lrcp_const*(qcl_parc+qcf_parc) &
382  +gamma_dry*z_theta(ii,k-1)
383  dq_sat_par_dil = repsilon*l_const*qsat_lev_dil(ii) &
384  /(r*t_ref_dil(ii)**2)
385 
386  q_liq_parc = max( 0.0, ( qw_parc(ii,k) - qsat_lev_dil(ii) &
387  -dq_sat_par_dil*( sl_parc(ii,k)-gamma_dry*z_theta(ii,k)-t_ref_dil(ii)) &
388  ) / (1.0+lrcp_const*dq_sat_par_dil) )
389 
390 ! add on the dIfference in the environment's ql as calculated by the
391 ! UM cloud scheme (using some RH_CRIT value) and what it
392 ! would be If RH_CRIT=1. This THEN imitates partial condensation
393 ! in the parcel.
394 !
395  ql_parc_dil(ii,k) = q_liq_parc + qcl(ii,k) + qcf(ii,k) - q_liq_env
396  t_parc_dil(ii,k) = sl_parc(ii,k)-gamma_dry*z_theta(ii,k) &
397  +lrcp_const*ql_parc_dil(ii,k)
398 
399 ! May need to recalculate if T_parc is > Tm and T_ref < Tm
400 
401  IF (t_ref_dil(ii) <= tm.and.t_parc_dil(ii,k) > tm) THEN
402 
403 ! recalculate using corrected latent heats
404  lrcp_const_parc = lcrcp
405 
406  q_liq_parc = max( 0.0, ( qw_parc(ii,k) - qsat_lev_dil(ii) &
407  -dq_sat_par_dil*(sl_parc(ii,k)-gamma_dry*z_theta(ii,k)-t_ref_dil(ii)) &
408  ) / (1.0+lrcp_const_parc*dq_sat_par_dil) )
409 
410  ql_parc_dil(ii,k) = q_liq_parc + qcl(ii,k) &
411  + qcf(ii,k)- q_liq_env
412 
413 ! revised at parcel calculation
414 
415  t_parc_dil(ii,k)=sl_parc(ii,k)-gamma_dry*z_theta(ii,k) &
416  +lrcp_const_parc*ql_parc_dil(ii,k)
417 
418  END IF ! test on t_ref
419 
420  q_vap_parc=qw_parc(ii,k)-ql_parc_dil(ii,k)
421 
422 ! Water loading in plume ( option 0 water remains in plume)
423 
424  IF (dil_plume_water_load == 1) THEN
425 
426  ! water removed from parcel after condesation if > 0.001 kg/kg
427  IF (ql_parc_dil(ii,k) > 0.001) THEN
428  ql_parc_dil(ii,k) = 0.001
429  qw_parc(ii,k) = q_vap_parc + ql_parc_dil(ii,k)
430  END IF
431 
432  ELSE IF (dil_plume_water_load == 2) THEN
433 
434  ! water remaining depends on qsat environment max_qw_plume = 0.5*qsat_env(ii)
435  max_qw_plume = min(qlcrit,max_qw_plume) ! max
436  max_qw_plume = max(qlmin,max_qw_plume) ! min value IF (ql_parc_dil(ii,k) > max_qw_plume) THEN
437  ql_parc_dil(ii,k) = max_qw_plume
438  qw_parc(ii,k) = q_vap_parc + ql_parc_dil(ii,k)
439  END IF
440  END IF ! end test on dil_plume_water_load
441 
442  t_dens_parc_dil(ii,k)=t_parc_dil(ii,k)* &
443  (1.0+c_virtual*q_vap_parc-ql_parc_dil(ii,k))
444 
445 ! calculate dilute t_ref for next level
446  IF (k > 1 .and. k < wet_levels-1) THEN
447  z_pr = (z_theta(ii,k+1)-z_theta(ii,k)) &
448  /(z_theta(ii,k)-z_theta(ii,k-1))
449 
450  th_par = t_parc_dil(ii,k)/exner_theta_levels(ii,k)
451  th_ref_dil(ii) = th_par*(1.+z_pr) - th_par_km_dil(ii)*z_pr
452 ! Check new reference sensible
453  IF (th_ref_dil(ii) < 0.0) THEN
454  th_ref_dil(ii) = th_par_km_dil(ii)
455  END IF
456  IF (th_par > 0.0) THEN th_par_km_dil(ii) = th_par
457  END IF
458  END IF ! k level test
459 
460  END IF ! test on LCL
461  END IF ! test on L_dilute
462 
463  buoyancy(ii,k) = t_dens_parc(ii,k) - t_dens_env(ii,k)
464 
465  env_svl(ii,k) = t_dens_env(ii,k) + gamma_dry*z_theta(ii,k)
466 
467  par_svl(ii,k) = t_dens_parc(ii,k) + gamma_dry*z_theta(ii,k)
468 
469  IF (l_dilute) THEN
470  buoyancy_dil(ii,k) = t_dens_parc_dil(ii,k) - t_dens_env(ii,k)
471  par_svl_dil(ii,k) = t_dens_parc_dil(ii,k) + gamma_dry*z_theta(ii,k)
472  END IF
473 
474  END DO ! Loop over ii
475 !$OMP END DO
476 
477 !$OMP END PARALLEL! Gradient calculations
