182 ! 1. Initialise.
183 ! -
184  IF (lhook) CALL dr_hook('ASAD_FTOY',zhook_in,zhook_handle)
185  IF ( gonce ) THEN
186  gonce = .false.
187  istmin = 0
188  ift = 0
189  DO j = 1, jpspec
190  ilstmin(j) = 0
191  ilft(j) = 0
192  ENDDO
193 
194  DO j = 1, nstst
195  js = nlstst(j)
196  IF ( ctype(js) == jpfm ) THEN
197  ifam = moffam(js)
198  imaj = majors(ifam)
199  IF ( imaj /= js ) THEN
200  istmin = istmin + 1
201  ilstmin(istmin) = js
202  ENDIF
203  ELSEIF ( ctype(js) == jpif ) THEN
204  istmin = istmin + 1
205  ilstmin(istmin) = js
206  ift = ift + 1
207  ilft(ift) = js
208  ELSEIF ( ctype(js) == jpna ) THEN
209  istmin = istmin + 1
210  ilstmin(istmin) = js
211  ELSE
212  CALL umprint( '**** ASAD ERROR in ASAD_FTOY! ',src='asad_ftoy')
213  CALL umprint( 'ASAD_FTOY found an unexpected species type', &
214  src='asad_ftoy')
215  CALL umprint( 'in the species list nlstst ',src='asad_ftoy')
216  cmessage='Found unexpected species type'
217  CALL ereport('ASAD_FTOY',js,cmessage)
218  ENDIF
219  ENDDO
220  ENDIF ! End of IF (gonce) statement
221 
222 ! 1.1 Set concentrations/initialise species
223 
224 ! DEPENDS ON: asad_fyinit

226 
227 ! 1.2 If there are no family or steady species then exit
228 
229  IF ( nstst == 0) THEN IF (lhook) CALL dr_hook('ASAD_FTOY',zhook_out,zhook_handle)
230  RETURN

238  js = nlstst(j)
239  DO jl = 1, n_points
240  zy(jl,js) = 0.0
241  ENDDO
242  ENDDO
243 
244  IF (ofirst .AND. iter < 5) THEN
245  iter = 5
246  icode = -1
247  cmessage = 'iter too low on first call, resetting to 5'
248  CALL ereport('ASAD_FTOY',icode,cmessage)

253 ! 2. Calculate self-reacting terms
254 ! - -
255 ! DEPENDS ON: asad_fyself
256  IF ( ofirst ) CALL asad_fyself(n_points)
257 
258 ! 3. Calculate family members using previous ratios
259 ! - -- --
260  IF (iter == 0) THEN
261 ! DEPENDS ON: asad_fyfixr
262  CALL asad_fyfixr(n_points)
263  IF (lhook) CALL dr_hook('ASAD_FTOY',zhook_out,zhook_handle)
264  RETURN
265  ENDIF
266 
267 ! 4. Iterate family members (including in/out species)
268 ! - - -! and non-model steady-state species
269 ! DO jit = 1, iter
270 
271 ! check to see if a species has gone zero with a nonzero
272 ! production.
273 
274  IF ( jit > 1 ) THEN
275  DO j = 1, nstst
276  js = nlstst(j)
277  DO jl = 1, n_points
278  IF (abs(y(jl,js)) < peps .AND. &
279  prod(jl,js) > peps ) THEN
280  y(jl,js) = peps
281  ENDIF
282  ENDDO
283  ENDDO
284  ENDIF
285 
286 ! 4.1 Calculate production and loss terms.
287 ! Note that the effect of deposition (wet & dry) and
288 ! emissions are not included in the ratios calculation.
289 ! Regardless of whether the user has them turned on or not.
290 ! See method above.
291 
292 ! DEPENDS ON: asad_prls
293  CALL asad_prls( nl, istmin, ilstmin, gdepem )
294 
295 ! 4.2 Initialise ratios
296 
297  DO j = 1, nstst
298  js = nlstst(j)
299  DO jl = 1, n_points
300  ratio(jl,js) = 0.0
301  ENDDO
302  ENDDO
303 
304 ! 4.3 Compute species values for species in steady state
305 ! (minor members of family, 'SS' and 'FT' species)
306 ! N.B. because of the way asad computes slos, the y
307 ! value can go slightly negative during the quadratic
308 ! due to loss of precision. We forcibly fix it here.
309 ! Also note that we cannot permit y=0.0 during the ftoy
310 ! iteration because of the need to get 'sl'.
311 
312  DO j = 1, istmin
313  js = ilstmin(j)
314  DO jl = 1, n_points
315  IF ( y(jl,js) < peps ) THEN
316  sl = 0.0
317  ELSE
318  sl = slos(jl,js) / y(jl,js)
319  ENDIF
320  IF ( qa(jl,js) > peps ) THEN
321  zb(jl) = sl - qa(jl,js) * y(jl,js)
322  zc(jl) = prod(jl,js)
323  zd(jl) = zb(jl)*zb(jl) + 4.0*qa(jl,js)*zc(jl)
324  IF ( zd(jl) > 0.0 ) THEN
325  y(jl,js) = (zb(jl) - sqrt(zd(jl)))/(-2.0*qa(jl,js))
326  if ( y(jl,js) < 0.0 ) y(jl,js) = 10.0*peps
327  ELSE
328  y(jl,js) = zb(jl) / ( -2.0 * qa(jl,js) )
329  ENDIF
330  ELSE IF (qa(jl,js) <= peps .AND. sl > peps ) THEN
331  y(jl,js) = prod(jl,js) / sl
332  ELSE
333  y(jl,js) = 0.0
334  END IF
335  ENDDO
336  ENDDO
337 
338 ! 4.4 Now compute ratios for minor species members.
339 ! ** could possibly take out 1/y(imaj) and convert to '*'
340 
341  istart = nlmajmin(3)
342  iend = nlmajmin(4)
343  DO j = nlmajmin(3), nlmajmin(4)
344  js = nlmajmin(j)
345  iodd = nodd(js)
346  ifam = moffam(js)
347  imaj = 0
348  IF ( ifam /= 0 ) imaj = majors(ifam)
349  DO jl = 1, n_points
350  IF ( y(jl,imaj) > peps ) THEN
351  ratio(jl,js) = y(jl,js) / y(jl,imaj)
352  ratio(jl,imaj) = ratio(jl,imaj) + &
353  iodd * ratio(jl,js)
354  ELSE
355  ratio(jl,js) = 0.0
356  END IF
357  ENDDO
358  ENDDO
359 
360 ! 4.5 Now compute ratios from species of type 'FT' if
361 ! applicable. If this is the first call, we also set
362 ! whether the species is in or out of the family.
363 
364  DO j = 1, ift
365  js = ilft(j)
366  iodd = nodd(js)
367  ifam = moffam(js)
368  itr = madvtr(js)
369  imaj = majors(ifam)
370  IF ( ofirst .and. jit == 1 ) THEN
371  DO jl = 1, n_points
372  IF ( y(jl,js) > peps ) then
373  sl = slos(jl,js) / y(jl,js)
374  ELSE
375  sl = 0.0
376  ENDIF
377  linfam(jl,itr) = sl > zthresh
378  IF ( linfam(jl,itr) ) &
379  f(jl,ifam) = f(jl,ifam) + iodd*f(jl,itr)
380  ENDDO
381  END IF
382 
383  DO jl = 1, n_points
384  if ( linfam(jl,itr) ) then
385  if ( y(jl,imaj) > peps ) then
386  ratio(jl,js) = y(jl,js) / y(jl,imaj)
387  ratio(jl,imaj) = ratio(jl,imaj) + &
388  iodd * ratio(jl,js)
389  else
390  ratio(jl,js) = 0.0
391  endif
392  else
393  y(jl,js) = f(jl,itr)
394  endif
395  ENDDO
396  ENDDO
397 
398 ! 4.6 Finally compute ratio of major species to family and
399 ! hence set the minor family members.
400 
401  istart = nlmajmin(1)
402  iend = nlmajmin(2)
403  DO j = istart, iend
404  js = nlmajmin(j)
405  iodd = nodd(js)
406  ifam = moffam(js)
407  DO jl = 1, n_points
408  ratio(jl,js) = 1.0 / (iodd + ratio(jl,js))
409  y(jl,js) = f(jl,ifam) * ratio(jl,js)
410  ENDDO
411  ENDDO
412 
413  istart = nlmajmin(3)
414  iend = nlmajmin(4)
415  DO j = istart, iend
416  js = nlmajmin(j)
417  ifam = moffam(js)
418  imaj = majors(ifam)
419  IF ( ctype(js) /= jpif ) THEN
420  DO jl = 1, n_points
421  y(jl,js) = y(jl,imaj) * ratio(jl,js)
422  ENDDO
423  else
424  itr = madvtr(js)
425  DO jl = 1, n_points
426  if ( linfam(jl,itr) ) y(jl,js) = &
427  y(jl,imaj) * ratio(jl,js)
428  ENDDO
429  ENDIF
430  ENDDO
431 
432 ! 4.7 Test for convergence.
433 
434  gconv = .true.
435  DO j = 1,nstst
436  js = nlstst(j)
437  DO jl = 1, n_points
438  IF ( abs(y(jl,js)-zy(jl,js)) > ftol*y(jl,js) &
439  .AND. y(jl,js) > pmintnd(jl) ) gconv=.false.
440  ENDDO
441  IF ( .NOT. gconv ) EXIT
442  ENDDO
443  IF (gconv) THEN IF (lhook) CALL dr_hook('ASAD_FTOY',zhook_out,zhook_handle)
444  RETURN
445  END IF
446 
447  DO j = 1, nstst
448  js = nlstst(j)
449  DO jl = 1, n_points
450  zy(jl,js) = y(jl,js)
451  ENDDO
452  ENDDO
453 
454  ENDDO ! End of iterations
455 
456 
457 ! 9. Convergence achieved or max. iterations reached.
458 
459  IF (lhook) CALL dr_hook('ASAD_FTOY',zhook_out,zhook_handle)
460  RETURN
