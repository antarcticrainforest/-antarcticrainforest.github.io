165 SUBROUTINE cosp_gridbox_mean(Npoints,Ncolumns,Nlevels,zhalf,zu,lstdgrid,x,r, &
166  log_units)
167  IMPLICIT NONE
168  ! Input arguments
169  INTEGER,INTENT(IN) :: Npoints !# of grid points
170  INTEGER,INTENT(IN) :: Nlevels !# of grid levels
171  INTEGER,INTENT(IN) :: Ncolumns !# of columns
172  REAL,DIMENSION(Npoints),INTENT(IN) :: zhalf ! Height at bottom half level [m]
173  REAL,DIMENSION(Nlevels),INTENT(IN) :: zu ! Upper boundary of new levels [m]
174  LOGICAL,INTENT(IN) :: lstdgrid ! Using standard grid (40 cloudsat levels)?
175  REAL,DIMENSION(Npoints,Ncolumns,Nlevels),INTENT(IN) :: x
176  !x - Input variable to be averaged
177  LOGICAL,OPTIONAL,INTENT(IN) :: log_units ! Need to convert to linear units
178  ! Output
179  REAL,DIMENSION(Npoints,Nlevels),INTENT(OUT) :: r
180  ! Variable averaged over subcolumns
181 
182  ! Local variables
183  INTEGER :: i,j,k,N
184  LOGICAL :: lunits
185  REAL :: t
186 
187  lunits=.false.
188  IF (PRESENT(log_units)) lunits=log_units
189 
190  r = 0.0
191 
192  DO k=1,nlevels
193  DO i=1,npoints
194  IF (lstdgrid .AND. (zu(k) <= zhalf(i))) THEN
195  ! Level of standard grid below model bottom level
196  r(i,k) = r_undef
197  ELSE
198  n = 0 ! Counter of valid points
199  IF (lunits) THEN
200  DO j=1,ncolumns
201  t = x(i,j,k)
202  IF (t /= r_undef) THEN
203  t = 10.0**(t/10.0)
204  ELSE
205  t = 0.0
206  ENDIF
207  r(i,k) = r(i,k) + t
208  n = n+1
209  ENDDO
210  ! Divide and change units back
211  IF (n /=0 ) r(i,k) = r(i,k)/n
212  IF (r(i,k) <= 0.0) THEN
213  r(i,k) = r_undef
214  ELSE
215  r(i,k) = 10.0*log10(r(i,k))
216  ENDIF
217  ELSE ! lunits=.false
218  DO j=1,ncolumns
219  t = x(i,j,k)
220  IF (t /= r_undef) THEN
221  r(i,k) = r(i,k) + t
222  n = n+1
223  ENDIF
224  ENDDO
225  ! Compute mean
226  IF (n /=0 ) THEN
227  r(i,k) = r(i,k)/n
228  ELSE
229  r(i,k) = r_undef
230  ENDIF
231  ENDIF ! LUNITS
232  ENDIF
233  ENDDO
234  ENDDO
235 
236 END SUBROUTINE cosp_gridbox_mean

240 SUBROUTINE cosp_change_vertical_grid(Npoints,Ncolumns,Nlevels,zfull,zhalf,y, &
241  Nglevels,newgrid_bot,newgrid_top,r,log_units)
242  IMPLICIT NONE
243  ! Input arguments
244  INTEGER,INTENT(IN) :: Npoints !# of grid points
245  INTEGER,INTENT(IN) :: Nlevels !# of levels
246  INTEGER,INTENT(IN) :: Ncolumns !# of columns
247  REAL,DIMENSION(Npoints,Nlevels),INTENT(IN) :: zfull
248  ! Height at model levels [m] (Bottom of model layer)
249  REAL,DIMENSION(Npoints,Nlevels),INTENT(IN) :: zhalf
250  ! Height at half model levels [m] (Bottom of model layer)
251  REAL,DIMENSION(Npoints,Ncolumns,Nlevels),INTENT(IN) :: y
252  ! Variable to be changed to a different grid
253  INTEGER,INTENT(IN) :: Nglevels !# levels in the new grid
254  REAL,DIMENSION(Nglevels),INTENT(IN) :: newgrid_bot
255  ! Lower boundary of new levels [m]
256  REAL,DIMENSION(Nglevels),INTENT(IN) :: newgrid_top
257  ! Upper boundary of new levels [m]
258  LOGICAL,OPTIONAL,INTENT(IN) :: log_units
259  ! log units, need to convert to linear units
260  ! Output
261  REAL,DIMENSION(Npoints,Ncolumns,Nglevels),INTENT(OUT) :: r
262  ! Variable on new grid
263 
264  ! Local variables
265  INTEGER :: i,j,k
266  LOGICAL :: lunits
267  INTEGER :: l
268  REAL :: w ! Weight
269  REAL :: dbb, dtb, dbt, dtt
270  ! Distances between edges of both grids
271  INTEGER :: Nw ! Number of weights
272  REAL :: wt ! Sum of weights
273  REAL,DIMENSION(Nlevels) :: oldgrid_bot,oldgrid_top
274  ! Lower and upper boundaries of model grid
275  REAL :: yp ! Local copy of y at a particular point.
276  ! This allows for change of units.
277 
278  lunits=.false.
279  IF (PRESENT(log_units)) lunits=log_units
280 
281  r = 0.0
282 
283  DO i=1,npoints
284  ! Calculate tops and bottoms of new and old grids
285  oldgrid_bot = zhalf(i,:)
286  oldgrid_top(1:nlevels-1) = oldgrid_bot(2:nlevels)
287  oldgrid_top(nlevels) = zfull(i,nlevels) + zfull(i,nlevels) - &
288  zhalf(i,nlevels) ! Top level symmetric
289  l = 0 ! Index of level in the old grid
290  ! Loop over levels in the new grid
291  DO k = 1,nglevels
292  nw = 0 ! Number of weigths
293  wt = 0.0 ! Sum of weights
294  ! Loop over levels in the old grid and accumulate
295  ! total for weighted average
296  DO
297  l = l + 1
298  w = 0.0 ! Initialise weight to 0
299  ! Distances between edges of both grids
300  dbb = oldgrid_bot(l) - newgrid_bot(k)
301  dtb = oldgrid_top(l) - newgrid_bot(k)
302  dbt = oldgrid_bot(l) - newgrid_top(k)
303  dtt = oldgrid_top(l) - newgrid_top(k)
304  IF (dbt >= 0.0) EXIT ! Do next level in the new grid
305  IF (dtb > 0.0) THEN
306  IF (dbb <= 0.0) THEN
307  IF (dtt <= 0) THEN
308  w = dtb
309  ELSE
310  w = newgrid_top(k) - newgrid_bot(k)
311  ENDIF
312  ELSE
313  IF (dtt <= 0) THEN
314  w = oldgrid_top(l) - oldgrid_bot(l)
315  ELSE
316  w = -dbt
317  ENDIF
318  ENDIF
319  ! If layers overlap (w/=0), then accumulate
320  IF (w /= 0.0) THEN
321  nw = nw + 1
322  wt = wt + w
323  DO j=1,ncolumns
324  IF (lunits) THEN
325  IF (y(i,j,l) /= r_undef) THEN
326  yp = 10.0**(y(i,j,l)/10.0)
327  ELSE
328  yp = 0.0
329  ENDIF
330  ELSE
331  yp = y(i,j,l)
332  ENDIF
333  r(i,j,k) = r(i,j,k) + w*yp
334  ENDDO
335  ENDIF
336  ENDIF
337  ENDDO
338  l = l - 2
339  IF (l < 1) l = 0
340  ! Calculate average in new grid
341  IF (nw > 0) THEN
342  DO j=1,ncolumns
343  r(i,j,k) = r(i,j,k)/wt
344  ENDDO
345  ENDIF
346  ENDDO
347  ENDDO
348 
349  ! Set points under surface to R_UNDEF, and change to dBZ if necessary
350  DO k=1,nglevels
351  DO j=1,ncolumns
352  DO i=1,npoints
353  IF (newgrid_top(k) > zhalf(i,1)) THEN ! Level above model bottom level
354  IF (lunits) THEN
355  IF (r(i,j,k) <= 0.0) THEN
356  r(i,j,k) = r_undef
357  ELSE
358  r(i,j,k) = 10.0*log10(r(i,j,k))
359  ENDIF
360  ENDIF
361  ELSE ! Level below surface
362  r(i,j,k) = r_ground
363  ENDIF
364  ENDDO
365  ENDDO
366  ENDDO
367 
368 END SUBROUTINE cosp_change_vertical_grid
