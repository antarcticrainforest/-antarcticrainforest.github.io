239 ! For each direction and observing level we calculate the
240 ! contribution of the particular integral to the radiance
241 ! from the surface and appropriate weightings for the
242 ! complementary function.
243 
244  DO id=1, n_direction
245 
246 ! Collect upward directions.
247  n_list_up=0
248  DO l=1, n_profile
249  IF (mu_v(l, id) > 0.0e+00_realk) THEN
250  n_list_up=n_list_up+1
251  list_up(n_list_up)=l
252  END IF
253  END DO

255 ! Calculate the angular arrays related to the BRDF. At higher
256 ! azimuthal orders there will be contributions because all
257 ! terms of the BRDF that would contribute are beyond the
258 ! level of truncation and so are zero.
259 
260  IF (ms <= ls_brdf_trunc-mod(ms, 2)) THEN
261 
262  DO j=1, n_brdf_basis_fnc
263  DO ls=ms, ls_trunc
264  lsr=ls-ms+1
265  DO ls_d=ms, ls_brdf_trunc-mod(ms, 2), 2
266  lsr_d=ls_d-ms+1
267  x=0.0e+00_realk
268  DO ls_dd=ms, ls_brdf_trunc-mod(ms, 2), 2
269  lsr_dd=ls_dd-ms+1
270  x=x-cgk((lsr_dd+1)/2, lsr)*f_brdf(j, ls_d, ls_dd, ms)
271  END DO
272  IF (ls_d == ms) THEN
273 ! Initialize this time.
274  DO l=1, n_profile
275  xy(l)=x*up_lm(l, lsr_d, id)
276  END DO
277  ELSE
278 ! Now add the increments.
279  DO l=1, n_profile
280  xy(l)=xy(l)+x*up_lm(l, lsr_d, id)
281  END DO
282  END IF
283  END DO
284  IF (j == 1) THEN

288  END DO

291  DO l=1, n_profile
292  ryx(l, lsr)=ryx(l, lsr)+rho_alb(l, j)*xy(l)

405  *brdf_full(l)
406  END DO
407 
408  ELSE IF (isolir == ip_infra_red) THEN
409  DO ll=1, n_list_up
410  l=list_up(ll)
411  brdf_full(l)=rho_alb(l, 1)*brdf_hemi(l, 1, id)
412  END DO
413  DO j=2, n_brdf_basis_fnc
414  DO ll=1, n_list_up
415  l=list_up(ll)
416  brdf_full(l)=brdf_full(l) &
417  +rho_alb(l, j)*brdf_hemi(l, j, id)
418  END DO
419  END DO
420  DO ll=1, n_list_up
421  l=list_up(ll)
422  radiance(l, ir, id)=radiance(l, ir, id) &
423  +trans(l) &
424  *(1.0e+00_realk-brdf_full(l)) &
425  *d_planck_flux_surface(l)/pi
426  END DO
427  END IF
428  END IF
429 
430  END DO
