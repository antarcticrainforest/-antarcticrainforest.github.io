14 FUNCTION eg_phi(xi2,xi3,a,eps)
15 USE ereport_mod, ONLY : ereport
16 IMPLICIT NONE
17 REAL, PARAMETER :: tol = 1.0e-10
18 INTEGER, PARAMETER :: it_max = 100
19 REAL :: eg_phi
20 REAL, INTENT(IN) :: xi2, xi3, a, eps
21 
22 REAL :: err, phi, cxi2, sxi2, z
23 REAL :: tmp, ep2
24 INTEGER :: it
25 
26 phi = abs(sin(xi2))
27 cxi2 = cos(xi2)**2
28 sxi2 = sin(xi2)**2
29 z = xi3/a
30 ep2 = eps**2
31 
32 IF( ep2 > 1.0e-8 ) THEN
33  DO it = 1, it_max
34  tmp = phi*z*sqrt( (1.0-ep2)/(1.0-ep2*phi**2) )
35  tmp = tmp**(2.0*ep2)
36 
37  tmp = sqrt( 1.0 - cxi2/( cxi2 + sxi2*tmp/(1.0-ep2) ) )
38  err = abs(tmp-phi)
39  phi = tmp
40  IF( err < tol ) EXIT
41  END DO
42 
43  IF( err > tol ) THEN
44  CALL ereport('eg_set_metric_terms',1, &
45  'Covergence Failure calculating phi in setup')
46  END IF
47  eg_phi = asin(phi)*sign(1.0,xi2)
48 ELSE
49  eg_phi = xi2
50 END IF
51 
52 END FUNCTION eg_phi
