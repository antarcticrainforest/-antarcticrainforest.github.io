114 ! calculate weight to eliminate a_minus_x on this row of the matrix
115  factor_forward(i,j,k) = - a_minus_x(i,j,k) / &
116  & a_central_x(i-1,j,k)
117 ! update central matrix entry for this row
118  a_central_x(i,j,k) = a_central_x(i,j,k) + &
119  & factor_forward(i,j,k) * &
120  & a_plus_x(i-1,j,k)
121 ! plus matrix entry is unchanged
122 ! storage space for minus matrix entry is now used to hold value for
123 ! zeroth point
124  a_minus_x(i,j,k) = factor_forward(i,j,k) * &
125  & a_minus_x(i-1,j,k)
126  End Do
127  End Do
128  End Do
129 
130 ! set-up reciprocal of a_central_x at this point as a_central_x is
131 ! not changed by any subsequent code. Reciprocal removes divisions
132 ! from code.
133  Do k = 1, model_levels
134  Do j = j_start, j_end
135  Do i = 1, row_length
136  recip_a_central_x(i,j,k) = 1. / a_central_x(i,j,k)
137  End Do
138  End Do
139  End Do

144  Do k = 1, model_levels

147 ! calculate weight to eliminate a_plus_x on this row of the matrix
148  factor_backward(i,j,k) = - a_plus_x(i,j,k) * &
149  & recip_a_central_x(i+1,j,k)
150 ! central matrix entry for this row is unchanged as minus entry on
151 ! row below has been removed by forward sweep.
152 ! plus matrix entry is now used to hold value for row_length+1 th
153 ! point.
154  a_plus_x(i,j,k) = factor_backward(i,j,k) * &
155  & a_plus_x(i+1,j,k)
156 ! a minus matrix entry is now used to hold value for
157 ! zeroth point and needs updating
158  a_minus_x(i,j,k) = a_minus_x(i,j,k) + &
159  & factor_backward(i,j,k) * &

161  End Do
162  End Do

167 
168 ! calculate position on row of processors, 0 = start, n_procx-1 = end

170  ime = me - ibase

172 ! number of boundary data points is twice number of processors

178  rbuf(1,j,k,ime) = a_minus_x(1,j,k)
179  rbuf(2,j,k,ime) = a_central_x(1,j,k)
180  rbuf(3,j,k,ime) = a_plus_x(1,j,k)
181  rbuf(4,j,k,ime) = a_minus_x(row_length,j,k)
182  rbuf(5,j,k,ime) = a_central_x(row_length,j,k)
183  rbuf(6,j,k,ime) = a_plus_x(row_length,j,k)
184  End Do
185  End Do

226  End Do
227

232  Do j = j_start, j_end
233 ! left hand boundary point on processor
234  bv_a_matrix(0,dj,j,k) = rbuf(2,j,k,0)
235  bv_a_matrix(2,dj,j,k) = rbuf(3,j,k,0)
236  bv_a_matrix_0(dj,j,k) = rbuf(1,j,k,0)
237 ! right hand boundary point on processor

239  bv_a_matrix(1,dj+1,j,k) = rbuf(6,j,k,0)

241  End Do

259  Do k = 1, model_levels
260  Do j = j_start, j_end
261 ! left hand boundary point on processor
262  bv_a_matrix(0,dj,j,k) = rbuf(2,j,k,d)
263  bv_a_matrix(2,dj,j,k) = rbuf(3,j,k,d)
264  bv_a_matrix(-1,dj,j,k) = rbuf(1,j,k,d)
265 ! right hand boundary point on processor
266  bv_a_matrix(0,dj+1,j,k) = rbuf(5,j,k,d)
267  bv_a_matrix(1,dj+1,j,k) = rbuf(6,j,k,d)

282  Do i = j_start, j_end
283  Do j = 2, nbv_points - 2
284  If (mod(j,2) == 0) Then
285 ! elimination from data on even rows
286 ! eliminate sub diagonal entry on row below
287  bv_factor_forward(1,j,i,k) = - bv_a_matrix(-1,j+1,i,k) / &
288  & bv_a_matrix(0,j,i,k)
289 ! set entry to zero
290  bv_a_matrix(-1,j+1,i,k) = 0.0
291 ! update other entries
292 ! 1. diagonal
293  bv_a_matrix(0,j+1,i,k) = bv_a_matrix(0,j+1,i,k) &
294  & + bv_factor_forward(1,j,i,k) &
295  & * bv_a_matrix(1,j,i,k)
296 ! 2. create 0 point entry
297  bv_a_matrix_0(j+1,i,k) = bv_factor_forward(1,j,i,k) &
298  & * bv_a_matrix_0(j,i,k)
299 ! eliminate sub-sub-diagonal entry on row 2 below
300  bv_factor_forward(2,j,i,k) = - bv_a_matrix(-2,j+2,i,k) / &
301  & bv_a_matrix(0,j,i,k)
302 ! set entry to zero
303 ! bv_a_matrix(-2,j+2,i,k) = 0.0
304 ! update other entries
305 ! 1. sub-diagonal (is zero to begin with)
306  bv_a_matrix(-1,j+2,i,k) = bv_factor_forward(2,j,i,k) &
307  & * bv_a_matrix(1,j,i,k)
308 ! 2. create 0 point entry

310  & * bv_a_matrix_0(j,i,k)
311 
312  Else

316  & bv_a_matrix(0,j,i,k)
317 ! set entry to zero

323  & + bv_factor_forward(1,j,i,k) &
324  & * bv_a_matrix(2,j,i,k)
325 ! 2. 0 point entry updating
326  bv_a_matrix_0(j+1,i,k) = bv_a_matrix_0(j+1,i,k) &
327  & + bv_factor_forward(1,j,i,k) &
328  & * bv_a_matrix_0(j,i,k)

389  End Do
