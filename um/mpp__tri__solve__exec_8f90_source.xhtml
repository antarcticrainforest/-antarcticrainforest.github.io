123 ! The following code snippet is the basic algorithm that has been optimised
124 ! for vector and scalar platforms below and is given for reference.
125 ! Do k = 1, model_levels ! Do j = j_start, j_end ! rhs_xx(1,j,k)=sss(1,j,k) ! End Do ! ! Do j = j_start, j_end ! 126 #if defined(VECTOR)
127 ! This version is best for NEC as it maintains a better vector length
128 ! Create a local copy of array sss to avoid bank conflicts
129 ! in the subsequent elimination sweep
130 ! The overhead of copying is much cheaper than the bank conflicts
131 ! However a better solution would be a thorough redesign of the
132 ! whole structure of the solver logic to avoid bad memory access
133 ! patterns by proper dimensioning of all arrays.
134  Do k = 1, model_levels
135  Do j = j_start, j_end
136  rhs_xx(1,j,k)=sss(1,j,k)
137  End Do
138  End Do
139 
140 !CDIR unroll=8
141  Do i = 2, row_length
142 !CDIR collapse
143  Do k = 1, model_levels
144  Do j = j_start, j_end
145 ! update right-hand-side term
146  rhs_xx(i,j,k) = sss(i,j,k) + factor_forward(i,j,k) * &
147  & rhs_xx(i-1,j,k)
148  End Do
149  End Do
150  End Do
151 
152 !CDIR unroll=8
153  Do i = row_length - 1, 1, -1
154 !CDIR collapse
155  Do k = 1, model_levels
156  Do j = j_start, j_end
157 ! update right-hand-side term
158  rhs_xx(i,j,k) = rhs_xx(i,j,k) + factor_backward(i,j,k) * &
159  & rhs_xx(i+1,j,k)
160  End Do
161  End Do
162  End Do
163 
164 #else
165 ! Hand unrolling by 4 with some
166 ! OpenMP added. Using the UNROLL directive doesn't seem to help unfortunately
167 !$OMP PARALLEL DO DEFAULT(NONE) SCHEDULE(STATIC) &
168 !$OMP& PRIVATE(i, j, k, jj, rhs_x_1, rhs_x_2, rhs_x_3, rhs_x_4) &
169 !$OMP& SHARED(model_levels, j_start, j_end, row_length, rhs_xx, sss, &
170 !$OMP& factor_forward, factor_backward)
171  Do k = 1, model_levels
172  Do j = j_start, j_end
173  rhs_xx(1,j,k)=sss(1,j,k)
174  End Do
175  Do j = j_start, j_end-3, 4
176 ! The rhs_x_(1-4) variables are the 4-strided rows of the first column
177 ! of rhs_xx at this point
178  rhs_x_1 = rhs_xx(1,j ,k)
179  rhs_x_2 = rhs_xx(1,j+1,k)
180  rhs_x_3 = rhs_xx(1,j+2,k)
181  rhs_x_4 = rhs_xx(1,j+3,k)
182 
183  Do i = 2, row_length
184 ! update right-hand-side term
185  rhs_x_1 = sss(i,j ,k) + factor_forward(i,j ,k) * rhs_x_1
186  rhs_x_2 = sss(i,j+1,k) + factor_forward(i,j+1,k) * rhs_x_2
187  rhs_x_3 = sss(i,j+2,k) + factor_forward(i,j+2,k) * rhs_x_3
188  rhs_x_4 = sss(i,j+3,k) + factor_forward(i,j+3,k) * rhs_x_4
189  rhs_xx(i,j ,k) = rhs_x_1
190  rhs_xx(i,j+1,k) = rhs_x_2
191  rhs_xx(i,j+2,k) = rhs_x_3
192  rhs_xx(i,j+3,k) = rhs_x_4
193  End Do
194 ! Now start the backward sweep - so start with data at end of rows
195 ! 4-strided again.
196  rhs_x_1 = rhs_xx(row_length,j ,k)
197  rhs_x_2 = rhs_xx(row_length,j+1,k)
198  rhs_x_3 = rhs_xx(row_length,j+2,k)
199  rhs_x_4 = rhs_xx(row_length,j+3,k)
200 
201  Do i = row_length - 1, 1, -1
202 ! update right-hand-side term
203  rhs_x_1 = rhs_xx(i,j ,k) + factor_backward(i,j ,k) * &
204  & rhs_x_1
205  rhs_x_2 = rhs_xx(i,j+1,k) + factor_backward(i,j+1,k) * &
206  & rhs_x_2
207  rhs_x_3 = rhs_xx(i,j+2,k) + factor_backward(i,j+2,k) * &
208  & rhs_x_3
209  rhs_x_4 = rhs_xx(i,j+3,k) + factor_backward(i,j+3,k) * &
210  & rhs_x_4
211  rhs_xx(i,j ,k) = rhs_x_1
212  rhs_xx(i,j+1,k) = rhs_x_2
213  rhs_xx(i,j+2,k) = rhs_x_3
214  rhs_xx(i,j+3,k) = rhs_x_4
215  End Do
216 
217  jj = j
218  End Do
219 
220 ! We've done the main calculation of rhs_xx, but as we've strided in
221 ! blocks of 4 there may be a few j elements not yet covered. These
222 ! will be mopped up starting after the finishing value in the previous
223 ! loop (stored in jj). The next value we need to compute is at jj+4
224 ! as the previous values have already been calculated in the strided loop.
225  jj = jj + 4
226  Do j=jj, j_end
227  Do i = 2, row_length
228  rhs_xx(i,j ,k) = sss(i,j ,k) + factor_forward(i,j ,k) * &
229  & rhs_xx(i-1,j ,k)
230  End Do
231 
232  Do i = row_length - 1, 1, -1
233  rhs_xx(i ,j,k) = rhs_xx(i ,j,k) + factor_backward(i ,j,k) * &
234  & rhs_xx(i+1,j ,k)
235  End Do
236  End Do
237  End Do

239 #endif

290 ! Section 3.1. Forward Gaussian elimination sweep on bv rhs
291 
292 !$OMP PARALLEL DEFAULT(NONE) &
293 !$OMP& PRIVATE(i, j, k, start_point, con1, con2) &
294 !$OMP& SHARED(j_start, j_end, nbv_points, model_levels, bv_rhs_x, &
295 !$OMP& bv_factor_forward, bv_factor_backward, bv_soln, &
296 !$OMP& bv_soln_1_term1, bv_soln_1_term2, bv_a_matrix_np1, &
297 !$OMP& bv_soln_n_term, bv_a_matrix_0, recip_bv_a_matrix_diag, &
298 !$OMP& soln, rhs_xx, a_minus_x, a_plus_x, recip_a_central_x, &

301 ! Note that in this parallel section we have many OMP Do loops with! NOWAIT clauses. The NOWAITs are only valid as we have a STATIC
302 ! SCHEDULE and the parallelised loops are over the same ranges in
303 ! all cases. This ensures a thread will get the same ranges of levels
304 ! throughout and dependencies between loops are thus satisfied.
305 !$OMP DO SCHEDULE(STATIC)
306  Do k = 1, model_levels
307  Do j = j_start, j_end
308  Do i = 2, nbv_points - 2, 2
309 ! elimination from data on even rows
310  bv_rhs_x(i+1,j,k) = bv_rhs_x(i+1,j,k) &
311  & + bv_factor_forward(1,i,j,k) &
312  & * bv_rhs_x(i,j,k)
313  bv_rhs_x(i+2,j,k) = bv_rhs_x(i+2,j,k) &
314  & + bv_factor_forward(2,i,j,k) &
315  & * bv_rhs_x(i,j,k)
316 
317 ! elimination from data on odd rows
318  bv_rhs_x(i+2,j,k) = bv_rhs_x(i+2,j,k) &
319  & + bv_factor_forward(1,i+1,j,k) &
320  & * bv_rhs_x(i+1,j,k)
321  End Do
322  End Do
323  End Do
324 !$OMP END DO NOWAIT
325 
326 
327 ! Section 3.2. Backward Gaussian elimination sweep on bv rhs
328 
329 
330 !$OMP DO SCHEDULE(STATIC)
331  Do k = 1, model_levels
332  Do j = j_start, j_end
333  Do i = nbv_points - 1, 3, -2
334  bv_rhs_x(i-2,j,k) = bv_rhs_x(i-2,j,k) &
335  & + bv_factor_backward(i,j,k) &
336  & * bv_rhs_x(i,j,k)
337  End Do
338  End Do
339  End Do
340 !$OMP END DO NOWAIT

342 
343 ! Section 3.3. Solution for end points 1 and nbv_points
344 
345 
346 !$OMP DO SCHEDULE(STATIC)
347  Do k = 1, model_levels
348  Do j = j_start, j_end
349  bv_soln(1,j,k) = ( bv_rhs_x(1,j,k) &
350  & - bv_soln_1_term1(j,k) * &
351  & bv_rhs_x(nbv_points,j,k) ) &
352  & * bv_soln_1_term2(j,k)
353  bv_soln(nbv_points,j,k) = ( bv_rhs_x(nbv_points,j,k) - &
354  & bv_a_matrix_np1(nbv_points,j,k) * &
355  & bv_soln(1,j,k) ) * &
356  & bv_soln_n_term(j,k)
357  End Do
358  End Do
359 !$OMP END DO NOWAIT
360 
361 ! set 0, and nbv_points+1 solutions, ie make solution periodic
362 !$OMP DO SCHEDULE(STATIC)
363  Do k = 1, model_levels
364  Do j = j_start, j_end
365  bv_soln(0,j,k) = bv_soln(nbv_points,j,k)
366  bv_soln(nbv_points+1,j,k) = bv_soln(1,j,k)
367  End Do
368  End Do
369 !$OMP END DO NOWAIT
370 
371 
372 ! Section 3.4. Find solution in inner of matrix
373 
374 
375 !$OMP DO SCHEDULE(STATIC)
376  Do k = 1, model_levels
377  Do j = j_start, j_end
378 ! first all odd rows
379  Do i = 3, nbv_points - 1, 2
380  bv_soln(i,j,k) = ( bv_rhs_x(i,j,k) &
381  & - bv_a_matrix_0(i,j,k) * bv_soln(0,j,k) &
382  & - bv_a_matrix_np1(i,j,k) * &
383  & bv_soln(nbv_points+1,j,k) ) &
384  & * recip_bv_a_matrix_diag(i,j,k)
385  End Do
386 ! now all even rows
387  Do i = 2, nbv_points - 2, 2
388  bv_soln(i,j,k) = ( bv_rhs_x(i,j,k) &
389  & - bv_a_matrix_sup(i,j,k) * bv_soln(i+1,j,k)&
390  & - bv_a_matrix_0(i,j,k) * bv_soln(0,j,k) &
391  & - bv_a_matrix_np1(i,j,k) * &
392  & bv_soln(nbv_points+1,j,k) ) &
393  & * recip_bv_a_matrix_diag(i,j,k)
394  End Do
395  End Do
396  End Do
397 !$OMP END DO NOWAIT
398 
399 
400 ! Section 3.5. Copy bv_soln into main solution
401 
402 
403  start_point = 2 * ime
404 !$OMP DO SCHEDULE(STATIC)
405  Do k = 1, model_levels
406  Do j = j_start, j_end
407  soln(0,j,k) = bv_soln(start_point,j,k)
408  soln(1,j,k) = bv_soln(start_point+1,j,k)
409  soln(row_length,j,k) = bv_soln(start_point+2,j,k)
410  soln(row_length+1,j,k) = bv_soln(start_point+3,j,k)
411  End Do
412  End Do
413 !$OMP END DO NOWAIT
414 
415 
416 ! Section 4. Solve for main solution not at boundaries
417 
418 
419 !$OMP DO SCHEDULE(STATIC)
420  Do k = 1, model_levels
421  Do j = j_start, j_end
422  con1=soln(0,j,k)
423  con2=soln(row_length+1,j,k)
424 !dir$ unroll
425  Do i = 2, row_length-1
426  soln(i,j,k) = ( rhs_xx(i,j,k) &
427  & - a_minus_x(i,j,k) * con1 &
428  & - a_plus_x(i,j,k) * con2 ) &
429  & * recip_a_central_x(i,j,k)
430  End Do
431  End Do
432  End Do
433 !$OMP END DO
434 !$OMP END PARALLEL
