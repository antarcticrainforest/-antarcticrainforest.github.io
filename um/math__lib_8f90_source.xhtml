180 !***********************************************************************
181 !
182 ! AVINT estimates the integral of unevenly spaced data.
183 !
184 ! Discussion:
185 !
186 ! The method uses overlapping parabolas and smoothing.
187 !
188 ! Modified:
189 !
190 ! 30 October 2000

192 ! loop to allow vectorization.
193 !
194 ! Reference:
195 !
196 ! Philip Davis and Philip Rabinowitz,
197 ! Methods of Numerical Integration,
198 ! Blaisdell Publishing, 1967.
199 !
200 ! P E Hennion,
201 ! Algorithm 77,
202 ! Interpolation, Differentiation and Integration,
203 ! Communications of the Association for Computing Machinery,
204 ! Volume 5, page 96, 1962.
205 !
206 ! Parameters:
207 !
208 ! Input, real ( kind = 8 ) FTAB(NTAB), the function values,
209 ! FTAB(I) = F(XTAB(I)).
210 !
211 ! Input, real ( kind = 8 ) XTAB(NTAB), the abscissas at which the
212 ! function values are given. The XTAB's must be distinct
213 ! and in ascending order.
214 !
215 ! Input, integer NTAB, the number of entries in FTAB and
216 ! XTAB. NTAB must be at least 3.

219 ! be, but need not be, near one endpoint of the interval
220 ! (X(1), X(NTAB)).
221 !
222 ! Input, real ( kind = 8 ) B, the upper limit of integration. B should

264  b = b_in

330 !
331 ! Carry out approximate integration from XTAB(ILO) to XTAB(IHI).
332 !
333  sum1 = 0.0d+00
334 
335  do i = ilo, ihi

339  x3 = xtab(i+1)
340 
341  term1 = ftab(i-1) / ( ( x1 - x2 ) * ( x1 - x3 ) )
342  term2 = ftab(i) / ( ( x2 - x1 ) * ( x2 - x3 ) )
343  term3 = ftab(i+1) / ( ( x3 - x1 ) * ( x3 - x2 ) )

348  - ( x1 + x3 ) * term2 &
349  - ( x1 + x2 ) * term3

352

355  cb = btemp
356  cc = ctemp
357  else

359  cb = 0.5d+00 * ( btemp + cb )
360  cc = 0.5d+00 * ( ctemp + cc )
361  end if

363  sum1 = sum1 &
364  + ca * ( x2**3 - syl**3 ) / 3.0d+00 &
365  + cb * 0.5d+00 * ( x2**2 - syl**2 ) &
366  + cc * ( x2 - syl )
367 
368  ca = atemp
369  cb = btemp
370  cc = ctemp
371 
372  syl = x2
373 
374  end do
375 
376  result = sum1 &
377  + ca * ( b**3 - syl**3 ) / 3.0d+00 &
378  + cb * 0.5d+00 * ( b**2 - syl**2 ) &
379  + cc * ( b - syl )
380 !
381 ! Restore original values of A and B, reverse sign of integral
382 ! because of earlier switch.
383 !
384  if ( ind /= 1 ) then
385  ind = 1
386  syl = b
387  b = a
388  a = syl
389  result = -result
390  end if
391 
392  return
393  end subroutine avint
